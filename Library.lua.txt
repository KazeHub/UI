-- Virtual Module Registry for high-performance isolated namespace resolution
local KazeRuntime = {
    _modules = {},
    _cache = {}
}

function KazeRuntime:Define(path, factory)
    self._modules[path] = factory
end

function KazeRuntime:Require(path)
    if self._cache[path] then
        return self._cache[path]
    end
    local factory = self._modules[path]
    if not factory then
        error("[KazeRuntime] Failed to resolve module path: " .. tostring(path))
    end
    local result = factory(self)
    self._cache[path] = result
    return result
end

-- MODULE: UTILITY & METRICS
-- ==========================================
KazeRuntime:Define("Utility", function(runtime)
    local TextService = game:GetService("TextService")
    local HttpService = game:GetService("HttpService")
    
    local Utility = {}
    Utility.IconsMap = {}
    
    -- Attempt safe recovery of Lucide icons dictionary
    pcall(function()
        if game and game.HttpGet then
            local rawIcons = game:HttpGet("https://raw.githubusercontent.com/KazeHub/UI/refs/heads/main/Icons.lua.txt")
            if rawIcons then
                local s, res = pcall(function()
                    local func = loadstring(rawIcons)
                    return func and func()
                end)
                if not s or type(res) ~= "table" then
                    s, res = pcall(function() return HttpService:JSONDecode(rawIcons) end)
                end
                if type(res) == "table" then
                    for k, v in pairs(res) do 
                        Utility.IconsMap[string.lower(tostring(k))] = tostring(v) 
                    end
                end
            end
        end
    end)

    function Utility.FormatImage(id)
        if not id or id == "" then return "" end
        id = tostring(id)
        local lowerId = string.lower(id)
        if Utility.IconsMap[lowerId] then
            local mapped = Utility.IconsMap[lowerId]
            if not string.find(mapped, "rbxassetid://") and not string.find(mapped, "http") then
                local clean = string.gsub(mapped, "%D", "")
                if clean ~= "" then return "rbxassetid://" .. clean end
            end
            return mapped
        end
        
        if string.find(id, "rbxassetid://") or string.find(id, "http") then return id end
        local clean = string.gsub(id, "%D", "")
        if clean == "" then return id end
        return "rbxassetid://" .. clean
    end

    function Utility.GetTextDimensions(text, size, font, bounds)
        bounds = bounds or Vector2.new(10000, 10000)
        return TextService:GetTextSize(text, size, font, bounds)
    end
    
    function Utility.DeepCopy(orig)
        local orig_type = type(orig)
        local copy
        if orig_type == 'table' then
            copy = {}
            for orig_key, orig_value in next, orig, nil do
                copy[Utility.DeepCopy(orig_key)] = Utility.DeepCopy(orig_value)
            end
            setmetatable(copy, Utility.DeepCopy(getmetatable(orig)))
        else
            copy = orig
        end
        return copy
    end

    return Utility
end)

-- MODULE: PHYSICAL SPRING ENGINE (Dynamics Solver)
-- ==========================================
KazeRuntime:Define("Spring", function(runtime)
    local RunService = game:GetService("RunService")
    local Spring = {}
    
    local EPSILON = 0.0001
    
    local function createSingleSpring(val, damping, speed)
        return {
            x = val,
            v = 0,
            t = val,
            d = damping or 1,
            s = speed or 12
        }
    end
    
    local function updateSingleSpring(spring, dt)
        local d = spring.d
        local s = spring.s
        local x = spring.x
        local v = spring.v
        local t = spring.t
        
        local force = s * s * (t - x) - 2 * s * d * v
        v = v + force * dt
        x = x + v * dt
        
        spring.x = x
        spring.v = v
        return x
    end

    local SpringSolver = {}
    SpringSolver.__index = SpringSolver

    function SpringSolver.new(val, damping, speed)
        local self = setmetatable({}, SpringSolver)
        self.damping = damping or 0.8
        self.speed = speed or 15
        self.type = typeof(val)
        self.springs = {}
        
        if self.type == "number" then
            self.springs[1] = createSingleSpring(val, self.damping, self.speed)
        elseif self.type == "Vector2" then
            self.springs[1] = createSingleSpring(val.X, self.damping, self.speed)
            self.springs[2] = createSingleSpring(val.Y, self.damping, self.speed)
        elseif self.type == "Vector3" then
            self.springs[1] = createSingleSpring(val.X, self.damping, self.speed)
            self.springs[2] = createSingleSpring(val.Y, self.damping, self.speed)
            self.springs[3] = createSingleSpring(val.Z, self.damping, self.speed)
        elseif self.type == "Color3" then
            self.springs[1] = createSingleSpring(val.R, self.damping, self.speed)
            self.springs[2] = createSingleSpring(val.G, self.damping, self.speed)
            self.springs[3] = createSingleSpring(val.B, self.damping, self.speed)
        elseif self.type == "UDim2" then
            self.springs[1] = createSingleSpring(val.X.Scale, self.damping, self.speed)
            self.springs[2] = createSingleSpring(val.X.Offset, self.damping, self.speed)
            self.springs[3] = createSingleSpring(val.Y.Scale, self.damping, self.speed)
            self.springs[4] = createSingleSpring(val.Y.Offset, self.damping, self.speed)
        end
        
        return self
    end

    function SpringSolver:SetTarget(targetVal)
        if self.type == "number" then
            self.springs[1].t = targetVal
        elseif self.type == "Vector2" then
            self.springs[1].t = targetVal.X
            self.springs[2].t = targetVal.Y
        elseif self.type == "Vector3" then
            self.springs[1].t = targetVal.X
            self.springs[2].t = targetVal.Y
            self.springs[3].t = targetVal.Z
        elseif self.type == "Color3" then
            self.springs[1].t = targetVal.R
            self.springs[2].t = targetVal.G
            self.springs[3].t = targetVal.B
        elseif self.type == "UDim2" then
            self.springs[1].t = targetVal.X.Scale
            self.springs[2].t = targetVal.X.Offset
            self.springs[3].t = targetVal.Y.Scale
            self.springs[4].t = targetVal.Y.Offset
        end
    end
    
    function SpringSolver:GetValue()
        if self.type == "number" then
            return self.springs[1].x
        elseif self.type == "Vector2" then
            return Vector2.new(self.springs[1].x, self.springs[2].x)
        elseif self.type == "Vector3" then
            return Vector3.new(self.springs[1].x, self.springs[2].x, self.springs[3].x)
        elseif self.type == "Color3" then
            return Color3.new(
                math.clamp(self.springs[1].x, 0, 1),
                math.clamp(self.springs[2].x, 0, 1),
                math.clamp(self.springs[3].x, 0, 1)
            )
        elseif self.type == "UDim2" then
            return UDim2.new(
                self.springs[1].x,
                math.round(self.springs[2].x),
                self.springs[3].x,
                math.round(self.springs[4].x)
            )
        end
    end

    function SpringSolver:Update(dt)
        local active = false
        for _, s in ipairs(self.springs) do
            s.d = self.damping
            s.s = self.speed
            updateSingleSpring(s, dt)
            if math.abs(s.v) > EPSILON or math.abs(s.t - s.x) > EPSILON then
                active = true
            end
        end
        return active
    end
    
    -- High Performance Tracker using a single RunService.Heartbeat pipeline
    local activeSprings = {}
    
    RunService.Heartbeat:Connect(function(dt)
        local dtCap = math.min(dt, 0.1) -- Frame skip safeguard
        for instance, bindings in pairs(activeSprings) do
            for prop, solver in pairs(bindings) do
                local isActive = solver:Update(dtCap)
                instance[prop] = solver:GetValue()
                if not isActive then
                    bindings[prop] = nil
                end
            end
            if next(bindings) == nil then
                activeSprings[instance] = nil
            end
        end
    end)
    
    function Spring.Animate(instance, prop, target, damping, speed)
        if not activeSprings[instance] then
            activeSprings[instance] = {}
        end
        
        local currentSolver = activeSprings[instance][prop]
        if not currentSolver then
            currentSolver = SpringSolver.new(instance[prop], damping, speed)
            activeSprings[instance][prop] = currentSolver
        else
            currentSolver.damping = damping or currentSolver.damping
            currentSolver.speed = speed or currentSolver.speed
        end
        
        currentSolver:SetTarget(target)
    end
    
    function Spring.Stop(instance, prop)
        if activeSprings[instance] then
            activeSprings[instance][prop] = nil
            if next(activeSprings[instance]) == nil then
                activeSprings[instance] = nil
            end
        end
    end

    return Spring
end)

-- MODULE: REACTIVE THEME SYSTEM (State Observers)

KazeRuntime:Define("Theme", function(runtime)
    local Theme = {}
    
    Theme.Presets = {
        Obsidian = {
            Name = "Obsidian",
            Background = Color3.fromRGB(10, 10, 14),
            Text = Color3.fromRGB(245, 245, 250),
            MutedText = Color3.fromRGB(205, 205, 215),
            Border = Color3.fromRGB(45, 45, 55),
            LightnessFactor = 1,
            BackgroundImage = "rbxassetid://13963496525",
            BackgroundImageTransparency = 0.85,
            OverlayTransparency = 0.20
        },
        CustomWallpaper = {
            Name = "Custom",
            Background = Color3.fromRGB(12, 12, 14),
            Text = Color3.fromRGB(255, 255, 255),
            MutedText = Color3.fromRGB(215, 210, 225),
            Border = Color3.fromRGB(55, 55, 60),
            LightnessFactor = 1,
            BackgroundImage = "",
            BackgroundImageTransparency = 0.1,
            OverlayTransparency = 0.65
        }
    }
    
    -- Local Engine State
    Theme.Current = Theme.Presets.Obsidian
    Theme.GlowColor = Color3.fromRGB(255, 255, 255)
    Theme.Transparency = 0.15
    
    local themeObservers = {}
    local glowObservers = {}
    
    -- Subscribes nodes securely to active state modifications without physical leak arrays
    function Theme.SubscribeTheme(instance, callback)
        themeObservers[instance] = callback
        callback(Theme.Current)
        
        instance.Destroying:Connect(function()
            themeObservers[instance] = nil
        end)
    end
    
    function Theme.SubscribeGlow(instance, callback)
        glowObservers[instance] = callback
        callback(Theme.GlowColor)
        
        instance.Destroying:Connect(function()
            glowObservers[instance] = nil
        end)
    end
    
    function Theme.SetTheme(nameOrPreset)
        local targetPreset
        if type(nameOrPreset) == "table" then
            targetPreset = nameOrPreset
        elseif type(nameOrPreset) == "string" then
            -- v3.5 Backward Compatibility: Parse numeric string or asset paths directly as CustomWallpaper
            local isAssetId = tonumber(nameOrPreset) ~= nil 
                or string.find(nameOrPreset, "rbxassetid://") 
                or string.find(nameOrPreset, "http")
            
            if isAssetId then
                Theme.Presets.CustomWallpaper.BackgroundImage = nameOrPreset
                targetPreset = Theme.Presets.CustomWallpaper
                Theme.Transparency = 0.65 -- Optimize alpha balance for image backgrounds
            else
                targetPreset = Theme.Presets[nameOrPreset] or Theme.Presets.Obsidian
            end
        else
            targetPreset = Theme.Presets.Obsidian
        end
        
        Theme.Current = targetPreset
        
        if targetPreset.Name == "Obsidian" then
            Theme.SetGlow(Color3.fromRGB(255, 255, 255))
        end
        
        for instance, cb in pairs(themeObservers) do
            task.spawn(cb, targetPreset)
        end
    end
    
    function Theme.SetGlow(color)
        Theme.GlowColor = color
        for instance, cb in pairs(glowObservers) do
            task.spawn(cb, color)
        end
    end
    
    function Theme.SetTransparency(alpha)
        Theme.Transparency = alpha
        Theme.SetTheme(Theme.Current) -- Force recalculations in panels
    end
    
    function Theme.GetElevationColor(elevation)
        local base = Theme.Current.Background
        local h, s, v = Color3.toHSV(base)
        local factor = Theme.Current.LightnessFactor or 1
        local offset = elevation * factor
        return Color3.fromHSV(h, s, math.clamp(v + offset, 0, 1))
    end
    
    function Theme.GetElevationTransparency(elevation)
        local baseTrans = Theme.Transparency
        if elevation > 0 then
            return math.clamp(baseTrans - (elevation * 0.25), 0.05, 1)
        elseif elevation < 0 then
            return math.clamp(baseTrans + (math.abs(elevation) * 0.25), 0.05, 1)
        end
        return baseTrans
    end

    return Theme
end)

KazeRuntime:Define("Effects", function(runtime)
    local Spring = runtime:Require("Spring")
    local Effects = {}
    
    -- Custom Dynamic Reactive Shadow Layer Generation
    function Effects.AddDropShadow(parent, cornerRadius)
        local shadow = Instance.new("Frame")
        shadow.Name = "KazeDropShadow"
        shadow.Size = UDim2.new(1, 12, 1, 12)
        shadow.Position = UDim2.fromOffset(-6, -6)
        shadow.BackgroundTransparency = 1
        shadow.ZIndex = -1
        shadow.Parent = parent
        
        local corner = Instance.new("UICorner", shadow)
        corner.CornerRadius = cornerRadius or UDim.new(0, 14)
        
        local stroke = Instance.new("UIStroke", shadow)
        stroke.Thickness = 4
        stroke.Color = Color3.fromRGB(0, 0, 0)
        stroke.Transparency = 0.8
        stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
        
        return shadow
    end
    
    -- Programmatic Interactive Circular Click Ripple Effect
    function Effects.CreateRipple(frame, position)
        local ripple = Instance.new("Frame")
        ripple.Name = "Ripple"
        ripple.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        ripple.BackgroundTransparency = 0.8
        ripple.AnchorPoint = Vector2.new(0.5, 0.5)
        ripple.ZIndex = frame.ZIndex + 1
        ripple.Parent = frame
        
        local corner = Instance.new("UICorner", ripple)
        corner.CornerRadius = UDim.new(1, 0)
        
        local originX = position.X - frame.AbsolutePosition.X
        local originY = position.Y - frame.AbsolutePosition.Y
        ripple.Position = UDim2.fromOffset(originX, originY)
        ripple.Size = UDim2.fromOffset(0, 0)
        
        local maxSize = math.max(frame.AbsoluteSize.X, frame.AbsoluteSize.Y) * 1.5
        
        Spring.Animate(ripple, "Size", UDim2.fromOffset(maxSize, maxSize), 0.8, 10)
        Spring.Animate(ripple, "BackgroundTransparency", 1, 0.8, 10)
        
        task.delay(1, function()
            ripple:Destroy()
        end)
    end
    
    -- Seamless Breathing Pulsation Animation Handler
    local breathingConnections = {}
    
    function Effects.StartBreathingGlow(stroke, baseColorSource, activeGlowSource)
        local isHovered = false
        
        local function updateGlow()
            if isHovered then
                Spring.Animate(stroke, "Color", activeGlowSource(), 0.7, 12)
            else
                Spring.Animate(stroke, "Color", baseColorSource(), 0.7, 12)
            end
        end
        
        local parent = stroke.Parent
        if parent and parent:IsA("GuiObject") then
            parent.MouseEnter:Connect(function()
                isHovered = true
                updateGlow()
            end)
            parent.MouseLeave:Connect(function()
                isHovered = false
                updateGlow()
            end)
        end
        
        -- Safe GC tracking using dynamic bindings
        breathingConnections[stroke] = updateGlow
        stroke.Destroying:Connect(function()
            breathingConnections[stroke] = nil
        end)
    end

    return Effects
end)

-- MODULE: INPUT ISOLATION ENGINE

KazeRuntime:Define("Input", function(runtime)
    local UIS = game:GetService("UserInputService")
    local Players = game:GetService("Players")
    local LP = Players.LocalPlayer
    local Mouse = LP:GetMouse()
    
    local Input = {}
    Input.ActiveResizing = false
    Input.ActiveDragging = false
    
    -- Multi-touch decoupled cursor and interaction tracker
    local CameraLocker = nil
    
    local function initLocker()
        if CameraLocker then return end
        CameraLocker = Instance.new("TextButton")
        CameraLocker.Name = "KazeCameraLocker"
        CameraLocker.Size = UDim2.fromOffset(0, 0)
        CameraLocker.BackgroundTransparency = 1
        CameraLocker.Text = ""
        CameraLocker.Modal = false
        CameraLocker.Visible = true
        
        local screen = LP:WaitForChild("PlayerGui", 10):FindFirstChild("KazeUI")
        if screen then CameraLocker.Parent = screen end
    end
    
    function Input.LockCamera()
        initLocker()
        if CameraLocker then CameraLocker.Modal = true end
    end
    
    function Input.UnlockCamera()
        initLocker()
        if CameraLocker then CameraLocker.Modal = false end
    end
    
    -- Premium multi-axis rigid dragging handler
    function Input.RegisterDrag(dragPart, targetPart)
        local Spring = runtime:Require("Spring")
        local dragging = false
        local dragInput, dragStart, startPos
        
        dragPart.Active = true
        
        dragPart.InputBegan:Connect(function(input)
            if Input.ActiveResizing then return end
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging = true
                Input.ActiveDragging = true
                dragStart = input.Position
                startPos = targetPart.Position
                
                Input.LockCamera()
                
                local conn
                conn = input.Changed:Connect(function()
                    if input.UserInputState == Enum.UserInputState.End then
                        dragging = false
                        Input.ActiveDragging = false
                        Input.UnlockCamera()
                        conn:Disconnect()
                    end
                end)
            end
        end)
        
        dragPart.InputChanged:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
                dragInput = input
            end
        end)
        
        UIS.InputChanged:Connect(function(input)
            if input == dragInput and dragging then
                local delta = input.Position - dragStart
                local scale = 1
                local screenScale = targetPart.Parent:FindFirstChildOfClass("UIScale")
                if screenScale then scale = screenScale.Scale end
                
                local nextPos = UDim2.new(
                    startPos.X.Scale, startPos.X.Offset + (delta.X / scale),
                    startPos.Y.Scale, startPos.Y.Offset + (delta.Y / scale)
                )
                
                Spring.Animate(targetPart, "Position", nextPos, 0.45, 20)
            end
        end)
    end

    return Input
end)

-- MODULE: PERSISTENT PROFILE CONFIGURATION
-- ==========================================
KazeRuntime:Define("Config", function(runtime)
    local HttpService = game:GetService("HttpService")
    local Config = {}
    
    local authorName = "Unknown"
    local flags = {}
    
    function Config.SetAuthor(name)
        authorName = tostring(name or "Unknown")
    end
    
    function Config.RegisterFlag(flag, getter, setter)
        if flag then
            flags[flag] = { Get = getter, Set = setter }
        end
    end
    
    function Config.Save(name, notifyCallback)
        if not name or name == "" then
            notifyCallback("Config Manager", "Configuration name cannot be blank.")
            return
        end
        
        local payload = {}
        for f, handler in pairs(flags) do
            payload[f] = handler.Get()
        end
        
        if writefile then
            local success, data = pcall(function() return HttpService:JSONEncode(payload) end)
            if success then
                if makefolder then
                    pcall(function()
                        if isfolder and not isfolder(authorName) then makefolder(authorName)
                        elseif not isfolder then makefolder(authorName) end
                    end)
                end
                writefile(authorName .. "/" .. name .. ".json", data)
                notifyCallback("Config Saved", "Successfully compiled & saved settings profile: " .. name)
            else
                notifyCallback("Config Error", "Failed to encode visual parameters securely.")
            end
        else
            notifyCallback("Config Error", "Local executor environment lacks writefile capability.")
        end
    end
    
    function Config.Load(name, notifyCallback)
        if readfile then
            local s, data = pcall(function() return readfile(authorName .. "/" .. name .. ".json") end)
            if s then
                local s2, tbl = pcall(function() return HttpService:JSONDecode(data) end)
                if s2 and type(tbl) == "table" then
                    -- Load priority visual flags before interactive structural widgets
                    if tbl["KazeUI_Theme_Preset"] and flags["KazeUI_Theme_Preset"] then
                        flags["KazeUI_Theme_Preset"].Set(tbl["KazeUI_Theme_Preset"])
                    end
                    
                    for f, val in pairs(tbl) do
                        if f ~= "KazeUI_Theme_Preset" and flags[f] then
                            flags[f].Set(val)
                        end
                    end
                    notifyCallback("Config Loaded", "Loaded dynamic settings configuration: " .. name)
                else
                    notifyCallback("Config Error", "Selected configuration structure contains corrupted metadata.")
                end
            else
                notifyCallback("Config Error", "Target profile was not discovered.")
            end
        end
    end
    
    function Config.GetProfiles()
        local list = {}
        if makefolder then
            pcall(function()
                if isfolder and not isfolder(authorName) then makefolder(authorName)
                elseif not isfolder then makefolder(authorName) end
            end)
        end
        
        if listfiles then
            local s, files = pcall(function() return listfiles(authorName) end)
            if s and type(files) == "table" then
                for _, file in ipairs(files) do
                    local cleanName = string.match(file, "([^/\\]+)$") or file
                    local core = string.match(cleanName, "^(.+)%.json$")
                    if core then table.insert(list, core) end
                end
            end
        end
        if #list == 0 then table.insert(list, "Default") end
        return list
    end
    
    function Config.Delete(name, notifyCallback)
        if delfile then
            pcall(function() delfile(authorName .. "/" .. name .. ".json") end)
            notifyCallback("Config Deleted", "Successfully removed setting configuration profile.")
        else
            notifyCallback("Config Error", "Executor does not support deletion operation.")
        end
    end

    return Config
end)

-- MODULE: COMPONENT SUBSYSTEMS

KazeRuntime:Define("Components", function(runtime)
    local Theme = runtime:Require("Theme")
    local Spring = runtime:Require("Spring")
    local Effects = runtime:Require("Effects")
    local Input = runtime:Require("Input")
    local Config = runtime:Require("Config")
    
    local Components = {}
    
    -- Sub-helper to safely bypass scrolling bounds during dropdown interactions
    local function toggleAncestorScroll(element, enabled)
        local cur = element.Parent
        while cur do
            if cur:IsA("ScrollingFrame") then
                cur.ScrollingEnabled = enabled
            end
            cur = cur.Parent
        end
    end

    -- CORE WIDGET: Collapsible Section
    function Components.CreateCollapsibleSection(parentFrame, title)
        local isOpen = false

        Theme.SubscribeTheme(boxStroke, function(theme)
            boxStroke.Color = theme.Border
        end)
        
        Theme.SubscribeTheme(sectionBox, function(t)
            sectionBox.BackgroundColor3 = Theme.GetElevationColor(0.02)
            sectionBox.BackgroundTransparency = Theme.GetElevationTransparency(0.02)

        local header = Instance.new("Frame")

        header.BackgroundTransparency = 1
        header.Parent = sectionBox

        local accent = Instance.new("Frame", header)

    -- CORE WIDGET: Toggles
    function Components.CreateToggle(parent, title, desc, default, callback, flag)
        local state = default or false
        
        local toggleFrame = Instance.new("Frame")
        toggleFrame.Size = UDim2.new(0.95, 0, 0, 52)
        toggleFrame.Parent = parent
        
        Instance.new("UICorner", toggleFrame).CornerRadius = UDim.new(0, 14)
        local stroke = Instance.new("UIStroke", toggleFrame)
        stroke.Thickness = 1.1
        
        Theme.SubscribeTheme(toggleFrame, function(t)
            toggleFrame.BackgroundColor3 = Theme.GetElevationColor(0.02)
            toggleFrame.BackgroundTransparency = Theme.GetElevationTransparency(0.02)
            stroke.Color = t.Border
        end)

        local label = Instance.new("TextLabel", toggleFrame)
        label.Size = UDim2.new(1, -80, 0, 20)
        label.Position = UDim2.fromOffset(14, 8)
        label.BackgroundTransparency = 1
        label.Text = title
        label.Font = Enum.Font.GothamBold
        label.TextSize = 13
        label.TextXAlignment = Enum.TextXAlignment.Left
        Theme.SubscribeTheme(label, function(t) label.TextColor3 = t.Text end)

        local description = Instance.new("TextLabel", toggleFrame)
        description.Size = UDim2.new(1, -80, 0, 16)
        description.Position = UDim2.fromOffset(14, 28)
        description.BackgroundTransparency = 1
        description.Text = desc
        description.Font = Enum.Font.Gotham
        description.TextSize = 11
        description.TextXAlignment = Enum.TextXAlignment.Left
        Theme.SubscribeTheme(description, function(t) description.TextColor3 = t.MutedText end)

        -- Switch layout track
        local track = Instance.new("Frame", toggleFrame)
        track.AnchorPoint = Vector2.new(1, 0.5)
        track.Position = UDim2.new(1, -14, 0.5, 0)
        track.Size = UDim2.fromOffset(38, 20)
        Instance.new("UICorner", track).CornerRadius = UDim.new(1, 0)
        
        local knob = Instance.new("Frame", track)
        knob.AnchorPoint = Vector2.new(0, 0.5)
        knob.Size = UDim2.fromOffset(14, 14)
        Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)
        knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)

        local clickBtn = Instance.new("TextButton", toggleFrame)
        clickBtn.Size = UDim2.fromScale(1, 1)
        clickBtn.BackgroundTransparency = 1
        clickBtn.Text = ""

        local function updateVisuals(instant)
            local targetPos = state and UDim2.new(1, -16, 0.5, 0) or UDim2.new(0, 2, 0.5, 0)
            
            if instant then
                knob.Position = targetPos
                if state then
                    Theme.SubscribeGlow(track, function(c) track.BackgroundColor3 = c end)
                else
                    track.BackgroundColor3 = Theme.GetElevationColor(0.08)
                end
            else
                Spring.Animate(knob, "Position", targetPos, 0.6, 18)
                if state then
                    Theme.SubscribeGlow(track, function(c) 
                        if state then Spring.Animate(track, "BackgroundColor3", c, 0.7, 15) end 
                    end)
                else
                    Spring.Animate(track, "BackgroundColor3", Theme.GetElevationColor(0.08), 0.7, 15)
                end
            end
        end
        
        updateVisuals(true)

        local function toggle()
            state = not state
            updateVisuals(false)
            if typeof(callback) == "function" then task.spawn(callback, state) end
        end

        clickBtn.MouseButton1Click:Connect(toggle)
        
        clickBtn.MouseEnter:Connect(function()
            Theme.SubscribeGlow(stroke, function(c) stroke.Color = c end)
        end)
        clickBtn.MouseLeave:Connect(function()
            Theme.SubscribeTheme(stroke, function(t) stroke.Color = t.Border end)
        end)

        Config.RegisterFlag(flag, function() return state end, function(val)
            state = val
            updateVisuals(false)
            if typeof(callback) == "function" then task.spawn(callback, state) end
        end)

        return {
            Frame = toggleFrame,
            SetState = function(self, s, inst) state = s; updateVisuals(inst) end,
            SetText = function(self, newTitle, newDesc)
                label.Text = newTitle or label.Text
                description.Text = newDesc or description.Text
            end
        }
    end

    -- CORE WIDGET: Numeric/Value Sliders
    function Components.CreateSlider(parent, title, min, max, default, callback, flag)
        local value = math.clamp(default or min, min, max)
        
        local sliderFrame = Instance.new("Frame")
        sliderFrame.Size = UDim2.new(0.95, 0, 0, 58)
        sliderFrame.Parent = parent
        
        Instance.new("UICorner", sliderFrame).CornerRadius = UDim.new(0, 14)
        local stroke = Instance.new("UIStroke", sliderFrame)
        stroke.Thickness = 1.1
        
        Theme.SubscribeTheme(sliderFrame, function(t)
            sliderFrame.BackgroundColor3 = Theme.GetElevationColor(0.02)
            sliderFrame.BackgroundTransparency = Theme.GetElevationTransparency(0.02)
            stroke.Color = t.Border
        end)

        local label = Instance.new("TextLabel", sliderFrame)
        label.Size = UDim2.new(1, -90, 0, 20)
        label.Position = UDim2.fromOffset(14, 8)
        label.BackgroundTransparency = 1
        label.Text = title
        label.Font = Enum.Font.GothamBold
        label.TextSize = 13
        label.TextXAlignment = Enum.TextXAlignment.Left
        Theme.SubscribeTheme(label, function(t) label.TextColor3 = t.Text end)

        local valLabel = Instance.new("TextLabel", sliderFrame)
        valLabel.Size = UDim2.new(0, 60, 0, 20)
        valLabel.Position = UDim2.new(1, -74, 0, 8)
        valLabel.BackgroundTransparency = 1
        valLabel.Text = tostring(value)
        valLabel.Font = Enum.Font.GothamBold
        valLabel.TextSize = 12
        valLabel.TextXAlignment = Enum.TextXAlignment.Right
        Theme.SubscribeTheme(valLabel, function(t) valLabel.TextColor3 = t.Text end)

        -- Track
        local track = Instance.new("Frame", sliderFrame)
        track.Size = UDim2.new(1, -28, 0, 6)
        track.Position = UDim2.new(0, 14, 0, 38)
        Instance.new("UICorner", track).CornerRadius = UDim.new(1, 0)
        
        Theme.SubscribeTheme(track, function()
            track.BackgroundColor3 = Theme.GetElevationColor(0.08)
        end)

        local fill = Instance.new("Frame", track)
        fill.Size = UDim2.fromScale(0, 1)
        Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)
        Theme.SubscribeGlow(fill, function(c) fill.BackgroundColor3 = c end)

        local knob = Instance.new("Frame", track)
        knob.AnchorPoint = Vector2.new(0.5, 0.5)
        knob.Size = UDim2.fromOffset(12, 12)
        knob.Position = UDim2.fromScale(0, 0.5)
        Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)
        knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)

        local clickBtn = Instance.new("TextButton", track)
        clickBtn.Size = UDim2.new(1, 0, 1, 14)
        clickBtn.Position = UDim2.new(0, 0, 0, -7)
        clickBtn.BackgroundTransparency = 1
        clickBtn.Text = ""

        local isDragging = false

        local function updateValue(input)
            local trackWidth = track.AbsoluteSize.X
            if trackWidth <= 0 then return end
            
            local percent = math.clamp((input.Position.X - track.AbsolutePosition.X) / trackWidth, 0, 1)
            local nextVal = math.round(min + (max - min) * percent)
            
            value = nextVal
            valLabel.Text = tostring(value)
            
            local visualPercent = (value - min) / (max - min)
            Spring.Animate(fill, "Size", UDim2.fromScale(visualPercent, 1), 0.5, 20)
            Spring.Animate(knob, "Position", UDim2.fromScale(visualPercent, 0.5), 0.5, 20)
            
            if typeof(callback) == "function" then task.spawn(callback, value) end
        end

        local function setValDirect(val)
            value = math.clamp(val, min, max)
            valLabel.Text = tostring(value)
            local visualPercent = (value - min) / (max - min)
            fill.Size = UDim2.fromScale(visualPercent, 1)
            knob.Position = UDim2.fromScale(visualPercent, 0.5)
            if typeof(callback) == "function" then task.spawn(callback, value) end
        end
        setValDirect(value)

        clickBtn.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                isDragging = true
                toggleAncestorScroll(track, false)
                Input.LockCamera()
                updateValue(input)
                
                Spring.Animate(knob, "Size", UDim2.fromOffset(16, 16), 0.5, 18)
                Theme.SubscribeGlow(stroke, function(c) stroke.Color = c end)
                
                local moveConn, endConn
                moveConn = game:GetService("UserInputService").InputChanged:Connect(function(mInput)
                    if isDragging and (mInput.UserInputType == Enum.UserInputType.MouseMovement or mInput.UserInputType == Enum.UserInputType.Touch) then
                        updateValue(mInput)
                    end
                end)
                endConn = game:GetService("UserInputService").InputEnded:Connect(function(eInput)
                    if eInput.UserInputType == Enum.UserInputType.MouseButton1 or eInput.UserInputType == Enum.UserInputType.Touch then
                        isDragging = false
                        toggleAncestorScroll(track, true)
                        Input.UnlockCamera()
                        
                        Spring.Animate(knob, "Size", UDim2.fromOffset(12, 12), 0.5, 18)
                        Theme.SubscribeTheme(stroke, function(t) stroke.Color = t.Border end)
                        moveConn:Disconnect()
                        endConn:Disconnect()
                    end
                end)
            end
        end)

        Config.RegisterFlag(flag, function() return value end, function(v) setValDirect(v) end)

        return {
            Frame = sliderFrame,
            SetValue = function(self, v) setValDirect(v) end,
            SetText = function(self, newTitle) label.Text = newTitle or label.Text end
        }
    end

        clickBtn.Size = UDim2.fromScale(1, 1)
        clickBtn.BackgroundTransparency = 1
        clickBtn.Text = ""

        local optionNodes = {}

        local function recalculateHeight()
            local count = 0
            for _, node in ipairs(optionNodes) do
                if node.Visible then count = count + 1 end
            end
            local targetListHeight = math.min(count, 5) * 32
            local totalHeight = isOpen and (90 + targetListHeight + 10) or 44
            
            Spring.Animate(dropFrame, "Size", UDim2.new(0.95, 0, 0, totalHeight), 0.7, 16)
            Spring.Animate(listScroller, "Size", UDim2.new(1, -28, 0, targetListHeight), 0.7, 16)
        end

        local function toggle()
            isOpen = not isOpen
            Spring.Animate(arrow, "Rotation", isOpen and 180 or 0, 0.7, 16)
            
            if isOpen then
                Theme.SubscribeGlow(stroke, function(c) if isOpen then stroke.Color = c end end)
            else
                Theme.SubscribeTheme(stroke, function(t) if not isOpen then stroke.Color = t.Border end end)
            end
            recalculateHeight()
        end
        clickBtn.MouseButton1Click:Connect(toggle)

        local function populate(opts)
            for _, v in ipairs(optionNodes) do v:Destroy() end
            table.clear(optionNodes)
            
            for index, value in ipairs(opts) do
                local optVal = tostring(value)
                local btn = Instance.new("TextButton")
                btn.Size = UDim2.new(1, 0, 0, 28)
                btn.Text = "  " .. optVal
                btn.Font = Enum.Font.Gotham
                btn.TextSize = 11
                btn.TextXAlignment = Enum.TextXAlignment.Left
                btn.Parent = listScroller
                Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)
                
                Theme.SubscribeTheme(btn, function()
                    local active = isMulti and selectedMulti[optVal] or (not isMulti and selectedSingle == optVal)
                    btn.BackgroundColor3 = active and Theme.GetElevationColor(0.08) or Theme.GetElevationColor(0.03)
                end)
                Theme.SubscribeTheme(btn, function(t)
                    local active = isMulti and selectedMulti[optVal] or (not isMulti and selectedSingle == optVal)
                    btn.TextColor3 = active and t.Text or t.MutedText
                end)

                local indicator
                if isMulti then
                    indicator = Instance.new("Frame", btn)
                    indicator.AnchorPoint = Vector2.new(1, 0.5)
                    indicator.Position = UDim2.new(1, -8, 0.5, 0)
                    indicator.Size = UDim2.fromOffset(12, 12)
                    Instance.new("UICorner", indicator).CornerRadius = UDim.new(0, 3)
                    
                    Theme.SubscribeTheme(indicator, function()
                        indicator.BackgroundColor3 = selectedMulti[optVal] and Theme.GlowColor or Theme.GetElevationColor(0.05)
                    end)
                end

                btn.MouseButton1Click:Connect(function()
                    if isMulti then
                        selectedMulti[optVal] = not selectedMulti[optVal]
                        Theme.SetTheme(Theme.Current) -- Force refresh
                        selectLabel.Text = getVisualText()
                        if typeof(callback) == "function" then
                            local ret = {}
                            for k, v in pairs(selectedMulti) do if v then table.insert(ret, k) end end
                            task.spawn(callback, ret)
                        end
                    else
                        selectedSingle = optVal
                        Theme.SetTheme(Theme.Current) -- Force refresh
                        selectLabel.Text = getVisualText()
                        toggle()
                        if typeof(callback) == "function" then task.spawn(callback, selectedSingle) end
                    end
                end)

                table.insert(optionNodes, btn)

        end)

            end

                    for _, v in ipairs(val) do selectedMulti[tostring(v)] = true end
                end
            else
                selectedSingle = tostring(val)
            end
            selectLabel.Text = getVisualText()
            Theme.SetTheme(Theme.Current)
        end)

        return {
            Frame = dropFrame,
            Refresh = function(self, newOpts) populate(newOpts) end,
            SetText = function(self, newText) label.Text = newText or label.Text end
        end

        btn.Position = UDim2.new(1, -14, 0.5, 0)
        btn.Size = UDim2.fromOffset(80, 24)
        Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
        
        local btnStroke = Instance.new("UIStroke", btn)
        btnStroke.Thickness = 1
        
        Theme.SubscribeTheme(btn, function(t)
            btn.BackgroundColor3 = Theme.GetElevationColor(0.05)
            btnStroke.Color = t.Border
        end)

        local bindLabel = Instance.new("TextLabel", btn)
        bindLabel.Size = UDim2.fromScale(1, 1)
        bindLabel.BackgroundTransparency = 1
        bindLabel.Text = bind == Enum.KeyCode.Unknown and "None" or bind.Name
        bindLabel.Font = Enum.Font.GothamBold
        bindLabel.TextSize = 10
        Theme.SubscribeTheme(bindLabel, function(t) bindLabel.TextColor3 = t.Text end)

        local function startListen()
            if listening then return end
            listening = true
            bindLabel.Text = "..."
            Theme.SubscribeGlow(btnStroke, function(c) btnStroke.Color = c end)
            
            local connection
            connection = game:GetService("UserInputService").InputBegan:Connect(function(input)
                local isKey = input.UserInputType == Enum.UserInputType.Keyboard
                local isMouse = input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.MouseButton2 or input.UserInputType == Enum.UserInputType.MouseButton3
                
                if isKey or isMouse then
                    if isKey then
                        if input.KeyCode == Enum.KeyCode.Escape then
                        elseif input.KeyCode == Enum.KeyCode.Backspace or input.KeyCode == Enum.KeyCode.Delete then
                            bind = Enum.KeyCode.Unknown
                        else
                            bind = input.KeyCode
                        end
                    end
                    bindLabel.Text = bind == Enum.KeyCode.Unknown and "None" or bind.Name
                    Theme.SubscribeTheme(btnStroke, function(t) btnStroke.Color = t.Border end)
                    connection:Disconnect()
                    task.delay(0.1, function() listening = false end)
                end
            end)
        end
        btn.MouseButton1Click:Connect(startListen)

        game:GetService("UserInputService").InputBegan:Connect(function(input, gp)
            if not gp and not listening and bind ~= Enum.KeyCode.Unknown and input.KeyCode == bind then
                if typeof(callback) == "function" then task.spawn(callback, bind) end
            end
        end)

        Config.RegisterFlag(flag, function() return bind.Name end, function(val)
            pcall(function()
                bind = Enum.KeyCode[val] or Enum.KeyCode.Unknown
                bindLabel.Text = bind == Enum.KeyCode.Unknown and "None" or bind.Name
            end)
        end)

        return {
            Frame = bindFrame,
            SetKey = function(self, key) bind = key; bindLabel.Text = bind == Enum.KeyCode.Unknown and "None" or bind.Name end
        }
    end

                local title = type(arg1) == "table" and arg1.Title or arg1

                local cb = type(arg1) == "table" and arg1.Callback or arg3
                return Components.CreateButton(targetParent, title, desc, cb)

                local t, d, def, cb, flag
                if type(arg1) == "table" then

                    t, d, def, cb, flag = arg1, arg2, arg3, arg4, arg5

                return Components.CreateToggle(targetParent, t, d, def, cb, flag)
            end

                    t, min, max, def, cb, flag = arg1, arg2, arg3, arg4, arg5, arg6

                return Components.CreateTextBox(targetParent, t, place, def, cb, flag)
            end
            
            function api:Keybind(arg1, arg2, arg3, arg4)
                local t, def, cb, flag
                if type(arg1) == "table" then
                    t, def, cb, flag = arg1.Title, arg1.Default, arg1.Callback, arg1.Flag
                else
                    t, def, cb, flag = arg1, arg2, arg3, arg4
                end
                return Components.CreateKeybind(targetParent, t, def, cb, flag)
            end
            
            function api:ColorPicker(arg1, arg2, arg3, arg4)
                local t, def, cb, flag
                if type(arg1) == "table" then
                    t, def, cb, flag = arg1.Title, arg1.Default, arg1.Callback, arg1.Flag
                else
                    t, def, cb, flag = arg1, arg2, arg3, arg4
                end
                return Components.CreateColorPicker(targetParent, t, def, cb, flag)
            end
            
            function api:Section(arg1)
                local t = type(arg1) == "table" and arg1.Title or arg1
                return Components.CreateDivider(targetParent, t)
            end
            
            function api:SectionUI(arg1)
                local t = type(arg1) == "table" and arg1.Title or arg1
                local sec = Components.CreateCollapsibleSection(targetParent, t)
                local secAPI = {}
                registerContainerElements(secAPI, sec.ContentFrame)
                return secAPI
            end
        end

        local windowAPI = {}
        
        function windowAPI:CreateTab(arg1, arg2)
            local tabTitle = type(arg1) == "table" and (arg1.Title or arg1.Name) or tostring(arg1 or "Tab")
            local tabIcon = type(arg1) == "table" and arg1.Icon or arg2
            
            local tabBtn = Instance.new("TextButton", tabScroller)
            tabBtn.Size = UDim2.new(1, 0, 0, 36)
            tabBtn.BackgroundTransparency = 1
            tabBtn.Text = ""
            Instance.new("UICorner", tabBtn).CornerRadius = UDim.new(0, 10)
            
            local leftInd = Instance.new("Frame", tabBtn)
            leftInd.Size = UDim2.new(0, 3, 0.6, 0)
            leftInd.Position = UDim2.new(0, 2, 0.2, 0)
            leftInd.BorderSizePixel = 0
            leftInd.Visible = false
            Instance.new("UICorner", leftInd).CornerRadius = UDim.new(1, 0)
            Theme.SubscribeGlow(leftInd, function(c) leftInd.BackgroundColor3 = c end)

            local iconLabel = Instance.new("ImageLabel", tabBtn)
            iconLabel.Size = UDim2.fromOffset(18, 18)
            iconLabel.Position = UDim2.fromOffset(12, 9)
            iconLabel.BackgroundTransparency = 1
            iconLabel.Image = Utility.FormatImage(tabIcon or "house")
            Theme.SubscribeTheme(iconLabel, function(t) iconLabel.ImageColor3 = t.MutedText end)

            local tLabel = Instance.new("TextLabel", tabBtn)
            tLabel.Size = UDim2.new(1, -40, 1, 0)
            tLabel.Position = UDim2.fromOffset(36, 0)
            tLabel.BackgroundTransparency = 1
            tLabel.Text = tabTitle
            tLabel.Font = Enum.Font.Gotham
            tLabel.TextSize = 12
            tLabel.TextXAlignment = Enum.TextXAlignment.Left
            Theme.SubscribeTheme(tLabel, function(t) tLabel.TextColor3 = t.MutedText end)

            local page = Instance.new("ScrollingFrame", containerPages)
            page.Size = UDim2.fromScale(1, 1)
            page.BackgroundTransparency = 1
            page.BorderSizePixel = 0
            page.Visible = false
            page.ScrollBarThickness = 2
            page.CanvasSize = UDim2.fromScale(0, 0)
            page.AutomaticCanvasSize = Enum.AutomaticSize.Y
            Theme.SubscribeTheme(page, function(t) page.ScrollBarImageColor3 = t.Border end)
            
            local pageLayout = Instance.new("UIListLayout", page)
            pageLayout.SortOrder = Enum.SortOrder.LayoutOrder
            pageLayout.Padding = UDim.new(0, 6)
            pageLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
            
            local pagePadding = Instance.new("UIPadding", page)
            pagePadding.PaddingTop = UDim.new(0, 12)
            pagePadding.PaddingBottom = UDim.new(0, 12)

            -- Tab visual state reconciliation
            local function selectTab()
                if activeTab then
                    activeTab.LeftInd.Visible = false
                    activeTab.Page.Visible = false
                    Spring.Animate(activeTab.Btn, "BackgroundTransparency", 1, 0.6, 12)
                    Theme.SubscribeTheme(activeTab.Label, function(t) activeTab.Label.TextColor3 = t.MutedText end)
                    Theme.SubscribeTheme(activeTab.Icon, function(t) activeTab.Icon.ImageColor3 = t.MutedText end)
                    activeTab.Label.Font = Enum.Font.Gotham
                end
                
                activeTab = { Btn = tabBtn, LeftInd = leftInd, Page = page, Label = tLabel, Icon = iconLabel }
                leftInd.Visible = true
                page.Visible = true
                Spring.Animate(tabBtn, "BackgroundTransparency", 0.9, 0.6, 12)
                tLabel.Font = Enum.Font.GothamBold
                
                Theme.SubscribeTheme(tabBtn, function()
                    tabBtn.BackgroundColor3 = Theme.GetElevationColor(0.04)
                end)
                Theme.SubscribeGlow(iconLabel, function(c) iconLabel.ImageColor3 = c end)
                Theme.SubscribeTheme(tLabel, function(t) tLabel.TextColor3 = t.Text end)
            end

            tabBtn.MouseButton1Click:Connect(selectTab)
            
            tabBtn.MouseEnter:Connect(function()
                if activeTab and activeTab.Btn == tabBtn then return end
                Spring.Animate(tabBtn, "BackgroundTransparency", 0.94, 0.6, 12)
                Theme.SubscribeTheme(tabBtn, function()
                    tabBtn.BackgroundColor3 = Theme.GetElevationColor(0.02)
                end)
                Theme.SubscribeTheme(tLabel, function(t) tLabel.TextColor3 = t.Text end)
            end)
            
            tabBtn.MouseLeave:Connect(function()
                if activeTab and activeTab.Btn == tabBtn then return end
                Spring.Animate(tabBtn, "BackgroundTransparency", 1, 0.6, 12)
                Theme.SubscribeTheme(tLabel, function(t) tLabel.TextColor3 = t.MutedText end)
            end)

            if #tabs == 0 then
                selectTab()
            end
            
            local tabObj = { Btn = tabBtn, Page = page, LeftInd = leftInd, Label = tLabel, Icon = iconLabel }
            table.insert(tabs, tabObj)

            local tabAPI = {}
            registerContainerElements(tabAPI, page)
            return tabAPI
        end

        function windowAPI:CreateConfigManager(tab)
            local sec = tab:SectionUI("Configuration Controller")
            local activeProfile = "Default"

            local configDropdown
            local function refreshProfiles()
                if configDropdown then
                    configDropdown:Refresh(Config.GetProfiles())
                end
            end

            sec:TextBox("Profile Name", "Type name...", "Default", function(val)
                activeProfile = val
            end)

            sec:Button("Save Profile Configuration", "Saves dynamic UI values under folder: " .. Author, function()
                Config.Save(activeProfile, function(t, c)
                    Notification.Notify({ Title = t, Content = c })
                end)
                refreshProfiles()
            end)

            configDropdown = sec:Dropdown("Discovered Config Profiles", Config.GetProfiles(), "Default", function(val)
                activeProfile = val
            end)

            sec:Button("Load Selected Settings Profile", "Applies all visualized parameters instantly", function()
                Config.Load(activeProfile, function(t, c)
                    Notification.Notify({ Title = t, Content = c, Icon = "check" })
                end)
            end)

            sec:Button("Purge Profile file", "Permanently deletes chosen profile JSON", function()
                Config.Delete(activeProfile, function(t, c)
                    Notification.Notify({ Title = t, Content = c, Icon = "trash" })
                end)
                refreshProfiles()
            end)
        end

        function windowAPI:CreateThemeManager(tab)
            local sec = tab:SectionUI("UI Theming Preferences")
            
            local tNames = {}
            for k, _ in pairs(Theme.Presets) do table.insert(tNames, k) end
            table.sort(tNames)

            sec:Dropdown("Layout Base Theme Selection", tNames, Theme.Current.Name, function(theme)
                Theme.SetTheme(theme)
            end, false, "KazeUI_Theme_Preset")

            sec:ColorPicker("Accent Color Visualizer", Theme.GlowColor, function(c)
                Theme.SetGlow(c)
            end, "KazeUI_Theme_GlowColor")

            sec:Slider("Window Surface Alpha", 0, 10, math.floor(Theme.Transparency * 10), function(v)
                Theme.SetTransparency(v / 10)
            end, "KazeUI_Theme_BackgroundTransparency")

            local wp = tab:SectionUI("Interactive Live Wallpapers")
            wp:TextBox("Image Asset Source", "rbxassetid:// or blank", Theme.Current.BackgroundImage or "", function(val)
                Theme.Current.BackgroundImage = val
                Theme.SetTheme(Theme.Current)
            end, "KazeUI_Wallpaper_AssetID")

            wp:Slider("Wallpaper Transparency", 0, 100, math.floor(Theme.Current.BackgroundImageTransparency * 100), function(v)
                Theme.Current.BackgroundImageTransparency = v / 100
                Theme.SetTheme(Theme.Current)
            end, "KazeUI_Wallpaper_Transparency")

            wp:Slider("Overlay Shade Opacity", 0, 100, math.floor(Theme.Current.OverlayTransparency * 100), function(v)
                Theme.Current.OverlayTransparency = v / 100
                Theme.SetTheme(Theme.Current)
            end, "KazeUI_Wallpaper_OverlayDensity")
        end

        function windowAPI:Minimize() minimize() end
        function windowAPI:Show() restore() end
        function windowAPI:Maximize() end
        function windowAPI:Toggle() if isMinimized then restore() else minimize() end end

        if typeof(config.Callback) == "function" then
            task.spawn(config.Callback)
        end

        return windowAPI
    end

    return WindowAPI
end)

    __index = function(tbl, key)
        if key == "CurrentTheme" then return Theme.Current
        elseif key == "GlowColor" then return Theme.GlowColor
        elseif key == "BackgroundTransparency" then return Theme.Transparency
        end
        return nil
    end
})

end

    Theme.SetTransparency(val)
end
