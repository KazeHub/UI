local function PlaySFX(soundName, pitchMod)
	task.spawn(function()
		if not Sounds[soundName] then return end
		local sfx = Instance.new("Sound")
		sfx.SoundId = Sounds[soundName]
		sfx.Volume = 0.6
		sfx.PlaybackSpeed = pitchMod or (0.95 + math.random() * 0.1) -- Slight organic pitch variance
		sfx.SoundGroup = SoundGroup
		sfx.Parent = SoundService
		sfx:Play()
		sfx.Ended:Connect(function() sfx:Destroy() end)
		task.delay(3, function() if sfx then sfx:Destroy() end end) -- Failsafe cleanup
	end)
end

-- Asset Loader
local IconsMap = {}
local function FormatImage(id)
	if not id or id == "" then return "" end
	local s = tostring(id)
	if IconsMap[string.lower(s)] then return IconsMap[string.lower(s)] end
	if string.find(s, "rbxassetid://") or string.find(s, "http") then return s end
	local num = string.match(s, "%d+")
	return num and "rbxassetid://" .. num or s
end

local function RegisterFlag(flag, getFunc, setFunc)
	if flag then KazeUI.Flags[flag] = { Get = getFunc, Set = setFunc } end
end

function KazeUI.GetColor(elevation)
	local t = KazeUI.CurrentTheme
	local h, s, v = Color3.toHSV(t.Panel)
	return Color3.fromHSV(h, s, math.clamp(v + (elevation * t.LightnessFactor), 0, 1))
end

function KazeUI:AddPanel(inst, elevation, ignoreColor, ignoreTrans)
	table.insert(KazeUI.PanelElements, {Inst = inst, Elevation = elevation or 0, IgnoreColor = ignoreColor, IgnoreTrans = ignoreTrans})
	if inst and inst:IsA("GuiObject") then
		if not ignoreColor then inst.BackgroundColor3 = KazeUI.GetColor(elevation or 0) end
		if not ignoreTrans then inst.BackgroundTransparency = KazeUI.BackgroundTransparency end
		inst.Destroying:Connect(function()
			for i = #KazeUI.PanelElements, 1, -1 do
				if KazeUI.PanelElements[i].Inst == inst then table.remove(KazeUI.PanelElements, i) end
			end
		end)
	end
end

function KazeUI:RegisterText(inst, isMuted)
	if not inst then return end
	table.insert(isMuted and KazeUI.MutedTextElements or KazeUI.TextElements, inst)
	inst.TextColor3 = isMuted and KazeUI.CurrentTheme.MutedText or KazeUI.CurrentTheme.Text
	if inst:IsA("TextBox") then inst.PlaceholderColor3 = KazeUI.CurrentTheme.MutedText end
	inst.Destroying:Connect(function()
		local tbl = isMuted and KazeUI.MutedTextElements or KazeUI.TextElements
		for i = #tbl, 1, -1 do if tbl[i] == inst then table.remove(tbl, i) end end
	end)
end

function KazeUI:RegisterBorder(inst)
	if not inst then return end
	table.insert(KazeUI.BorderElements, inst)
	inst.Color = KazeUI.CurrentTheme.Border
	inst.Destroying:Connect(function()
		for i = #KazeUI.BorderElements, 1, -1 do if KazeUI.BorderElements[i] == inst then table.remove(KazeUI.BorderElements, i) end end
	end)
end

function KazeUI:OnThemeChanged(inst, cb)
	table.insert(KazeUI.ThemeCallbacks, {Inst = inst, Callback = cb})
	task.spawn(cb, KazeUI.CurrentTheme)
end

function KazeUI:OnGlowChanged(inst, cb)
	table.insert(KazeUI.GlowCallbacks, {Inst = inst, Callback = cb})
	task.spawn(cb, KazeUI.GlowColor)
end

function KazeUI:SetTheme(themeName)
	local target = KazeUI.Themes[themeName] or KazeUI.Themes.Obsidian
	KazeUI.CurrentTheme = target
	
	for _, data in ipairs(KazeUI.PanelElements) do
		if data.Inst and data.Inst.Parent and not data.IgnoreColor then data.Inst.BackgroundColor3 = KazeUI.GetColor(data.Elevation) end
	end
	for _, inst in ipairs(KazeUI.TextElements) do if inst.Parent then inst.TextColor3 = target.Text end end
	for _, inst in ipairs(KazeUI.MutedTextElements) do 
		if inst.Parent then inst.TextColor3 = target.MutedText
			if inst:IsA("TextBox") then inst.PlaceholderColor3 = target.MutedText end
		end 
	end
	for _, inst in ipairs(KazeUI.BorderElements) do if inst.Parent then inst.Color = target.Border end end
	
	local hasImage = target.BackgroundImage ~= ""
	for _, wp in ipairs(KazeUI.WallpaperElements) do
		if wp.ImageLabel and wp.ImageLabel.Parent then
			wp.ImageLabel.Image = FormatImage(target.BackgroundImage)
			TweenService:Create(wp.ImageLabel, TWEEN_MECH, {ImageTransparency = hasImage and target.BackgroundImageTransparency or 1}):Play()
			TweenService:Create(wp.OverlayFrame, TWEEN_MECH, {BackgroundTransparency = hasImage and target.OverlayTransparency or 1}):Play()
		end
	end
	for _, data in ipairs(KazeUI.ThemeCallbacks) do if data.Inst and data.Inst.Parent then pcall(data.Callback, target) end end
end

function KazeUI:SetTransparency(alpha)
	KazeUI.BackgroundTransparency = alpha
	for _, data in ipairs(KazeUI.PanelElements) do
		if data.Inst and data.Inst.Parent and not data.IgnoreTrans then data.Inst.BackgroundTransparency = alpha end
	end
end

function KazeUI:SetGlow(color)
	KazeUI.GlowColor = color
	for i = #KazeUI.GlowCallbacks, 1, -1 do
		local data = KazeUI.GlowCallbacks[i]
		if data.Inst and data.Inst.Parent then task.spawn(data.Callback, color) else table.remove(KazeUI.GlowCallbacks, i) end
	end
end

local function TypewriterEffect(label, text, speed)
	label.Text = ""
	local chars = string.split(text, "")
	for i = 1, #chars do
		label.Text = label.Text .. chars[i]
		if i % 3 == 0 then PlaySFX("Hover", 1.5) end
		task.wait(speed or 0.02)
	end
end

local function GlitchEffect(element)
	local origPos = element.Position
	local origTrans = element.BackgroundTransparency
	for i = 1, 3 do
		element.Position = origPos + UDim2.fromOffset(math.random(-2, 2), math.random(-2, 2))
		element.BackgroundTransparency = origTrans + 0.2
		task.wait(0.05)
	end
	element.Position = origPos
	element.BackgroundTransparency = origTrans
end

-- ==========================================
-- DRAG & RESIZE ENGINE
-- ==========================================
local lastNormalPosition = UDim2.fromScale(0.5, 0.5)
local lastNormalAnchor = Vector2.new(0.5, 0.5)

local function MakeDraggable(dragPart, targetPart)
	local dragging, dragStart, startPos, dragInput
	dragPart.Active = true
	dragPart.InputBegan:Connect(function(input)
		if KazeUI.IsResizing then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true; dragStart = input.Position; startPos = targetPart.Position
			LockCamera()
			local conn; conn = input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false; UnlockCamera(); conn:Disconnect()
				end
			end)
		end
	end)
	dragPart.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
	end)
	UIS.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			local delta = input.Position - dragStart
			local s = UIScale.Scale
			targetPart.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + (delta.X / s), startPos.Y.Scale, startPos.Y.Offset + (delta.Y / s))
			if targetPart.Name == "HUD_Window" then lastNormalPosition = targetPart.Position; lastNormalAnchor = targetPart.AnchorPoint end
		end
	end)
end

function KazeUI:SaveConfig(name)
	if not name or name == "" then return self:Notify({Title = "SYS_ERROR", Content = "INVALID CONFIG NAME"}) end
	local data = {}
	for flag, obj in pairs(KazeUI.Flags) do data[flag] = obj.Get() end
	if writefile then
		local s, enc = pcall(function() return HttpService:JSONEncode(data) end)
		if s then
			if makefolder and (not isfolder or not isfolder(KazeUI.AuthorName)) then pcall(makefolder, KazeUI.AuthorName) end
			writefile(KazeUI.AuthorName .. "/" .. name .. ".json", enc)
			self:Notify({Title = "SYS_SAVED", Content = "DATA WRITTEN: " .. name})
			PlaySFX("Scan")
		end
	else
		self:Notify({Title = "SYS_ERROR", Content = "FS_WRITE_UNSUPPORTED"})
	end
end

function KazeUI:LoadConfig(name)
	if readfile then
		local s, res = pcall(function() return readfile(KazeUI.AuthorName .. "/" .. name .. ".json") end)
		if s then
			local s2, data = pcall(function() return HttpService:JSONDecode(res) end)
			if s2 and type(data) == "table" then
				if data["KazeUI_Theme_Preset"] and KazeUI.Flags["KazeUI_Theme_Preset"] then
					KazeUI.Flags["KazeUI_Theme_Preset"].Set(data["KazeUI_Theme_Preset"])
				end
				for flag, val in pairs(data) do
					if flag ~= "KazeUI_Theme_Preset" and KazeUI.Flags[flag] then KazeUI.Flags[flag].Set(val) end
				end
				self:Notify({Title = "SYS_LOADED", Content = "DATA RESTORED: " .. name})
				PlaySFX("Boot")
			end
		else
			self:Notify({Title = "SYS_ERROR", Content = "DATA NOT FOUND: " .. name})
		end
	end
end

	return wrapper, frame, stroke
end

	inner.Position = UDim2.fromOffset(10, 8)
	inner.BackgroundTransparency = 1
	
	local pTitle, pBody
	if tHeight > 0 then
		pTitle = Instance.new("TextLabel", inner)
		pTitle.Size = UDim2.new(1, -50, 0, tHeight)
		pTitle.BackgroundTransparency = 1
		pTitle.Text = title
		pTitle.Font = Enum.Font.RobotoMono
		pTitle.TextSize = 13
		pTitle.TextXAlignment = Enum.TextXAlignment.Left
		if bHeight == 0 then pTitle.AnchorPoint = Vector2.new(0, 0.5); pTitle.Position = UDim2.new(0, 0, 0.5, 0) end
		KazeUI:RegisterText(pTitle, false)
	end
	if bHeight > 0 then
		pBody = Instance.new("TextLabel", inner)
		pBody.Size = UDim2.new(1, -50, 0, bHeight)
		pBody.Position = UDim2.fromOffset(0, tHeight + 2)
		pBody.BackgroundTransparency = 1
		pBody.Text = body
		pBody.Font = Enum.Font.RobotoMono
		pBody.TextSize = 11
		pBody.TextXAlignment = Enum.TextXAlignment.Left
		pBody.TextWrapped = true
		KazeUI:RegisterText(pBody, true)
	end

	local box = Instance.new("Frame", inner)
	box.AnchorPoint = Vector2.new(1, 0.5)
	box.Position = UDim2.new(1, 0, 0.5, 0)
	box.Size = UDim2.fromOffset(24, 24)
	box.BackgroundColor3 = KazeUI.GetColor(0.02)
	box.BorderSizePixel = 0
	Instance.new("UICorner", box).CornerRadius = UDim.new(0, 2)

	check.AnchorPoint = Vector2.new(0.5, 0.5)
	check.Position = UDim2.fromScale(0.5, 0.5)

		local tColor = state and KazeUI.GlowColor or KazeUI.CurrentTheme.Border
		if instant then
			check.Size = state and UDim2.fromScale(0.6, 0.6) or UDim2.fromScale(0, 0)

	RegisterFlag(flag, function() return state end, function(val) state = val; setVisuals(false); if type(callback) == "function" then task.spawn(callback, state) end end)
	
	return { Wrapper = wrapper, SetState = function(_, v) state = v; setVisuals(false) end }

local function CreateSliderInput(parentFrame, title, min, max, default, callback, flag)
	local value = math.clamp(default or min, min, max)
	local wrapper, frame, stroke = CreateHUDFrame(parentFrame, 54)
	
	local inner = Instance.new("Frame", frame)
	inner.Size = UDim2.new(1, -20, 1, -16)
	inner.Position = UDim2.fromOffset(10, 8)
	inner.BackgroundTransparency = 1
	
	local pTitle = Instance.new("TextLabel", inner)
	pTitle.Size = UDim2.new(1, -80, 0, 18)
	pTitle.BackgroundTransparency = 1
	pTitle.Text = title
	pTitle.Font = Enum.Font.RobotoMono
	pTitle.TextSize = 13
	pTitle.TextXAlignment = Enum.TextXAlignment.Left
	KazeUI:RegisterText(pTitle, false)
	
	local valLabel = Instance.new("TextLabel", inner)
	valLabel.Size = UDim2.new(0, 70, 0, 18)
	valLabel.Position = UDim2.new(1, -70, 0, 0)
	valLabel.BackgroundTransparency = 1
	valLabel.Text = string.format("[ %d ]", value)
	valLabel.Font = Enum.Font.RobotoMono
	valLabel.TextSize = 12
	valLabel.TextXAlignment = Enum.TextXAlignment.Right
	KazeUI:RegisterText(valLabel, false)
	
	local track = Instance.new("Frame", inner)
	track.Size = UDim2.new(1, 0, 0, 4)
	track.Position = UDim2.new(0, 0, 0, 26)
	track.BorderSizePixel = 0
	KazeUI:AddPanel(track, -0.05)
	
	local fill = Instance.new("Frame", track)
	fill.Size = UDim2.new(0, 0, 1, 0)
	fill.BorderSizePixel = 0
	
	local knob = Instance.new("Frame", track)
	knob.AnchorPoint = Vector2.new(0.5, 0.5)
	knob.Size = UDim2.fromOffset(8, 16) -- Tall rectangular grip
	knob.Position = UDim2.new(0, 0, 0.5, 0)
	knob.BorderSizePixel = 0
	
	local function updateVisuals()
		local pct = max > min and (value - min) / (max - min) or 0
		valLabel.Text = string.format("[ %d ]", value)
		TweenService:Create(fill, TWEEN_FAST, {Size = UDim2.new(pct, 0, 1, 0)}):Play()
		TweenService:Create(knob, TWEEN_FAST, {Position = UDim2.new(pct, 0, 0.5, 0)}):Play()
	end
	updateVisuals()
	
	local isDragging = false
	local btn = Instance.new("TextButton", track)
	btn.Size = UDim2.new(1, 0, 1, 16); btn.Position = UDim2.fromOffset(0, -8)
	btn.BackgroundTransparency = 1; btn.Text = ""

	local function processDrag(input)
		local pct = math.clamp((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
		value = math.clamp(math.floor(min + (max - min) * pct + 0.5), min, max)
		updateVisuals()
		if type(callback) == "function" then task.spawn(callback, value) end
	end
	
	btn.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			isDragging = true; LockCamera(); processDrag(input); PlaySFX("Hover")
			TweenService:Create(knob, TWEEN_FAST, {Size = UDim2.fromOffset(10, 20)}):Play()
			
			local conMove, conEnd
			conMove = UIS.InputChanged:Connect(function(mInput)
				if isDragging and (mInput.UserInputType == Enum.UserInputType.MouseMovement or mInput.UserInputType == Enum.UserInputType.Touch) then
					processDrag(mInput)
				end
			end)
			conEnd = UIS.InputEnded:Connect(function(eInput)
				if eInput.UserInputType == Enum.UserInputType.MouseButton1 or eInput.UserInputType == Enum.UserInputType.Touch then
					isDragging = false; UnlockCamera(); TweenService:Create(knob, TWEEN_FAST, {Size = UDim2.fromOffset(8, 16)}):Play(); PlaySFX("Click")
					conMove:Disconnect(); conEnd:Disconnect()
				end
			end)
		end
	end)

	KazeUI:OnGlowChanged(fill, function(c) fill.BackgroundColor3 = c; knob.BackgroundColor3 = c end)
	RegisterFlag(flag, function() return value end, function(v) value = math.clamp(v, min, max); updateVisuals(); if type(callback) == "function" then task.spawn(callback, value) end end)
	
	return { Wrapper = wrapper, SetValue = function(_, v) value = math.clamp(v, min, max); updateVisuals() end }
end

local function CreateDropdown(parentFrame, title, options, defaultOption, callback, isMulti, flag)
	options = options or {}
	local selectedSingle = not isMulti and (defaultOption or options[1] or "") or ""
	local selectedMulti = {}
	if isMulti then
		if type(defaultOption) == "table" then for _, v in ipairs(defaultOption) do selectedMulti[tostring(v)] = true end
		elseif defaultOption then selectedMulti[tostring(defaultOption)] = true end
	end
	
	local isOpen = false
	local wrapper = Instance.new("Frame", parentFrame)
	wrapper.Size = UDim2.new(1, 0, 0, 0); wrapper.BackgroundTransparency = 1; wrapper.AutomaticSize = Enum.AutomaticSize.Y
	
	local frame = Instance.new("Frame", wrapper)
	frame.AnchorPoint = Vector2.new(0.5, 0); frame.Position = UDim2.new(0.5, 0, 0, 2)
	frame.Size = UDim2.new(0.96, 0, 0, 0); frame.BorderSizePixel = 0; frame.AutomaticSize = Enum.AutomaticSize.Y
	Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 2)
	KazeUI:AddPanel(frame, 0.05)
	
	local pad = Instance.new("UIPadding", frame); pad.PaddingTop = UDim.new(0, 5); pad.PaddingBottom = UDim.new(0, 5)
	local stroke = Instance.new("UIStroke", frame); stroke.Thickness = 1; KazeUI:RegisterBorder(stroke)
	
	local layout = Instance.new("UIListLayout", frame)
	layout.SortOrder = Enum.SortOrder.LayoutOrder; layout.HorizontalAlignment = Enum.HorizontalAlignment.Center; layout.Padding = UDim.new(0, 2)
	
	local head = Instance.new("Frame", frame); head.Size = UDim2.new(1, -20, 0, 32); head.BackgroundTransparency = 1; head.LayoutOrder = 1
	local pTitle = Instance.new("TextLabel", head); pTitle.Size = UDim2.new(0.5, 0, 1, 0); pTitle.BackgroundTransparency = 1
	pTitle.Text = string.upper(title); pTitle.Font = Enum.Font.RobotoMono; pTitle.TextSize = 13; pTitle.TextXAlignment = Enum.TextXAlignment.Left
	KazeUI:RegisterText(pTitle, false)
	
	local valLab = Instance.new("TextLabel", head); valLab.Size = UDim2.new(0.5, -20, 1, 0); valLab.Position = UDim2.new(0.5, 0, 0, 0)
	valLab.BackgroundTransparency = 1; valLab.Font = Enum.Font.RobotoMono; valLab.TextSize = 11; valLab.TextXAlignment = Enum.TextXAlignment.Right; valLab.TextTruncate = Enum.TextTruncate.AtEnd
	KazeUI:RegisterText(valLab, true)
	
	local arrow = Instance.new("TextLabel", head); arrow.AnchorPoint = Vector2.new(1, 0.5); arrow.Position = UDim2.new(1, 0, 0.5, 0); arrow.Size = UDim2.fromOffset(16, 16)
	arrow.BackgroundTransparency = 1; arrow.Text = "▼"; arrow.Font = Enum.Font.RobotoMono; arrow.TextSize = 12
	KazeUI:RegisterText(arrow, true)
	
	local function updateHead()
		if isMulti then
			local c=0; local f=""; for k,v in pairs(selectedMulti) do if v then c=c+1; if f=="" then f=k end end end
			valLab.Text = c==0 and "[ NONE ]" or (c==1 and "[ " .. f .. " ]" or "[ " .. c .. " SEL ]")
		else valLab.Text = "[ " .. tostring(selectedSingle) .. " ]" end
	end
	updateHead()
	
	local btn = Instance.new("TextButton", head); btn.Size = UDim2.fromScale(1,1); btn.BackgroundTransparency = 1; btn.Text = ""
	
	local listContainer = Instance.new("ScrollingFrame", frame)
	listContainer.Size = UDim2.new(1, -16, 0, 0); listContainer.BackgroundTransparency = 1; listContainer.BorderSizePixel = 0
	listContainer.LayoutOrder = 2; listContainer.Visible = false; listContainer.ScrollBarThickness = 2
	listContainer.ScrollBarImageColor3 = KazeUI.GlowColor
	local lLayout = Instance.new("UIListLayout", listContainer); lLayout.SortOrder = Enum.SortOrder.LayoutOrder; lLayout.Padding = UDim.new(0, 2)
	
	local optBtns = {}
	local function closeDropdown()
		isOpen = false; PlaySFX("Hover")
		TweenService:Create(arrow, TWEEN_MECH, {Rotation = 0}):Play()
		TweenService:Create(listContainer, TWEEN_MECH, {Size = UDim2.new(1, -16, 0, 0)}):Play()
		task.delay(0.2, function() if not isOpen then listContainer.Visible = false end end)
	end
	
	local function refresh(newOpts)
		for _, v in ipairs(optBtns) do v.Btn:Destroy() end; table.clear(optBtns)
		options = newOpts or {}
		for _, opt in ipairs(options) do
			local oStr = tostring(opt)
			local oBtn = Instance.new("TextButton", listContainer)
			oBtn.Size = UDim2.new(1, 0, 0, 26); oBtn.BorderSizePixel = 0; oBtn.Text = ""
			Instance.new("UICorner", oBtn).CornerRadius = UDim.new(0, 2); KazeUI:AddPanel(oBtn, 0.08, true)
			
			local oLab = Instance.new("TextLabel", oBtn)
			oLab.Size = UDim2.new(1, -10, 1, 0); oLab.Position = UDim2.fromOffset(10, 0); oLab.BackgroundTransparency = 1
			oLab.Text = oStr; oLab.Font = Enum.Font.RobotoMono; oLab.TextSize = 11; oLab.TextXAlignment = Enum.TextXAlignment.Left
			
			local function updateVis()
				local act = isMulti and selectedMulti[oStr] or (not isMulti and selectedSingle == oStr)
				oBtn.BackgroundColor3 = act and KazeUI.GetColor(0.12) or KazeUI.GetColor(0.04)
				oLab.TextColor3 = act and KazeUI.GlowColor or KazeUI.CurrentTheme.MutedText
				oLab.Text = act and "> " .. oStr or "  " .. oStr
			end
			updateVis()
			oBtn.MouseEnter:Connect(function() oBtn.BackgroundColor3 = KazeUI.GetColor(0.15); PlaySFX("Hover") end)
			oBtn.MouseLeave:Connect(updateVis)
			oBtn.MouseButton1Click:Connect(function()
				PlaySFX("Click")
				if isMulti then selectedMulti[oStr] = not selectedMulti[oStr]; updateVis(); updateHead()
					if type(callback) == "function" then local r={}; for k,v in pairs(selectedMulti) do if v then table.insert(r,k) end end; task.spawn(callback, r) end
				else selectedSingle = oStr; updateHead(); closeDropdown(); for _, b in ipairs(optBtns) do b.Upd() end
					if type(callback) == "function" then task.spawn(callback, selectedSingle) end
				end
			end)
			table.insert(optBtns, {Btn = oBtn, Upd = updateVis})
		end
		listContainer.CanvasSize = UDim2.new(0, 0, 0, #options * 28)
	end
	refresh(options)
	
	btn.MouseButton1Click:Connect(function()
		isOpen = not isOpen
		if isOpen then
			PlaySFX("Hover")
			listContainer.Visible = true
			local tgtH = math.min(#options, 5) * 28
			TweenService:Create(arrow, TWEEN_MECH, {Rotation = 180}):Play()
			TweenService:Create(listContainer, TWEEN_MECH, {Size = UDim2.new(1, -16, 0, tgtH)}):Play()
		else closeDropdown() end
	end)
	
	return { Wrapper = wrapper, Refresh = function(_, n) refresh(n) end }
end

local function AttachAPI(api, container)
	function api:AddLabel(arg)
		local t = type(arg) == "table" and (arg.Text or arg.Title) or arg
		local lab = Instance.new("TextLabel", container)
		lab.Size = UDim2.new(1, -12, 0, 20); lab.BackgroundTransparency = 1; lab.Text = tostring(t or "")
		lab.Font = Enum.Font.RobotoMono; lab.TextSize = 12; lab.TextXAlignment = Enum.TextXAlignment.Left; lab.TextWrapped = true
		KazeUI:RegisterText(lab, true)
		return lab
	end
	function api:Button(a1, a2, a3)
		local t = type(a1)=="table" and a1.Title or a1; local d = type(a1)=="table" and (a1.Description or a1.Content) or a2; local cb = type(a1)=="table" and a1.Callback or a3
		return CreateClickableParagraph(container, t, d or "", cb)
	end
	function api:Toggle(a1, a2, a3, a4, a5)
		local t, d, def, cb, fl = "", "", false, nil, nil
		if type(a1) == "table" then t,d,def,cb,fl = a1.Title, a1.Description or "", a1.Default, a1.Callback, a1.Flag
		else t,d,def,cb,fl = a1,a2,a3,a4,a5 end
		return CreateToggleSwitch(container, t, d, def, cb, fl)
	end
	function api:Slider(a1, a2, a3, a4, a5, a6)
		local t, min, max, def, cb, fl = "", 0, 100, 50, nil, nil
		if type(a1) == "table" then t,min,max,def,cb,fl = a1.Title, a1.Min, a1.Max, a1.Default, a1.Callback, a1.Flag
		else t,min,max,def,cb,fl = a1,a2,a3,a4,a5,a6 end
		return CreateSliderInput(container, t, min, max, def, cb, fl)
	end
	function api:Dropdown(a1, a2, a3, a4, a5, a6)
		local t, opts, def, mul, cb, fl = "", {}, nil, false, nil, nil
		if type(a1) == "table" then t,opts,def,mul,cb,fl = a1.Title, a1.Options or a1.Values, a1.Default, a1.Multi or false, a1.Callback, a1.Flag
		else t,opts,def,cb,mul,fl = a1,a2,a3,a4,a5 or false,a6 end
		return CreateDropdown(container, t, opts, def, cb, mul, fl)
	end
	function api:TextBox(a1, a2, a3, a4, a5)
		local t, pl, def, cb, fl = "", "", "", nil, nil
		if type(a1) == "table" then t,pl,def,cb,fl = a1.Title, a1.Placeholder, a1.Default, a1.Callback, a1.Flag
		else t,pl,def,cb,fl = a1,a2,a3,a4,a5 end
		return CreateTextBoxInput(container, t, pl, def, cb, fl)
	end
	function api:Section(a1) return CreateSectionLabel(container, type(a1)=="table" and a1.Title or a1) end
	-- Aliasing SectionUI to Section for mechanical theme (collapsible sections break mecha flow, standardizing to headers)
	function api:SectionUI(a1) return CreateSectionLabel(container, type(a1)=="table" and a1.Title or a1) end
	-- Fallbacks
	function api:Paragraph(a1, a2) return CreateClickableParagraph(container, type(a1)=="table" and a1.Title or a1, type(a1)=="table" and (a1.Description or a1.Content) or a2, nil) end
	function api:ColorPicker(a1) return api:TextBox({Title = type(a1)=="table" and a1.Title or a1, Placeholder = "HEX COLOR...", Callback = type(a1)=="table" and a1.Callback or nil, Flag = type(a1)=="table" and a1.Flag or nil}) end
	function api:Keybind(a1) return api:Button({Title = type(a1)=="table" and a1.Title or a1, Description = "[ KEYBIND SYSTEM UNAVAILABLE IN HUD MODE ]"}) end
end

	local WinAPI = { _Pages = PagesContainer, _Tabs = {} }
	
	function WinAPI:CreateTab(name)
		name = type(name)=="table" and (name.Title or name.Name) or name or "TAB_X"
		local btn = Instance.new("TextButton", TabsScroller)
		btn.Size = UDim2.new(1, 0, 0, 35)
		btn.BackgroundColor3 = Color3.fromRGB(15, 15, 17)
		btn.BorderSizePixel = 0
		btn.Text = ""
		
		local ind = Instance.new("Frame", btn)
		ind.Size = UDim2.new(0, 3, 1, 0)
		ind.BorderSizePixel = 0; ind.Visible = false
		KazeUI:OnGlowChanged(ind, function(c) ind.BackgroundColor3 = c end)
		
		local txt = Instance.new("TextLabel", btn)
		txt.Size = UDim2.new(1, -15, 1, 0)
		txt.Position = UDim2.fromOffset(15, 0)
		txt.BackgroundTransparency = 1
		txt.Text = "> " .. string.upper(name)
		txt.Font = Enum.Font.RobotoMono
		txt.TextSize = 12
		txt.TextXAlignment = Enum.TextXAlignment.Left
		KazeUI:RegisterText(txt, true)
		
		local page = Instance.new("ScrollingFrame", self._Pages)
		page.Size = UDim2.new(1, 0, 1, 0)
		page.BackgroundTransparency = 1
		page.BorderSizePixel = 0
		page.ScrollBarThickness = 2
		page.ScrollBarImageColor3 = KazeUI.GlowColor
		page.Visible = false
		local pPad = Instance.new("UIPadding", page); pPad.PaddingTop = UDim.new(0, 10); pPad.PaddingBottom = UDim.new(0, 10); pPad.PaddingLeft = UDim.new(0, 10); pPad.PaddingRight = UDim.new(0, 10)
		local pLay = Instance.new("UIListLayout", page); pLay.SortOrder = Enum.SortOrder.LayoutOrder; pLay.Padding = UDim.new(0, 6); pLay.HorizontalAlignment = Enum.HorizontalAlignment.Center
		
		-- Dynamic Canvas Sizer
		page.ChildAdded:Connect(function() task.wait(); page.CanvasSize = UDim2.new(0, 0, 0, pLay.AbsoluteContentSize.Y + 20) end)
		
		table.insert(self._Tabs, { Btn = btn, Ind = ind, Txt = txt, Page = page })
		
		local function activate()
			PlaySFX("Click")
			for _, t in ipairs(self._Tabs) do
				t.Ind.Visible = false; t.Page.Visible = false
				t.Btn.BackgroundColor3 = Color3.fromRGB(15, 15, 17)
				t.Txt.TextColor3 = KazeUI.CurrentTheme.MutedText
			end
			ind.Visible = true; page.Visible = true
			btn.BackgroundColor3 = Color3.fromRGB(20, 20, 22)
			txt.TextColor3 = KazeUI.GlowColor
			GlitchEffect(page)
		end
		btn.MouseButton1Click:Connect(activate)
		if #self._Tabs == 1 then activate() end
		
		local tabAPI = {}
		AttachAPI(tabAPI, page)
		return tabAPI
	end
	
	function WinAPI:CreateConfigManager(tab)
		tab:Section("CONFIG_SYS")
		local cfgName = "SYS_DEF"
		tab:TextBox("CONFIG_ID", "ENTER FILENAME", "SYS_DEF", function(v) cfgName = v end)
		tab:Button("EXEC_SAVE", "WRITE TO LOCAL FS", function() KazeUI:SaveConfig(cfgName) end)
		tab:Button("EXEC_LOAD", "READ FROM LOCAL FS", function() KazeUI:LoadConfig(cfgName) end)
	end
	
	function WinAPI:CreateThemeManager(tab)
		tab:Section("VISUAL_CORE")
		tab:TextBox("GLOW_HEX", "#00FFFF", "", function(v) pcall(function() KazeUI:SetGlow(Color3.fromHex(v)) end) end)
	end
