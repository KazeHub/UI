local function UpdateScale()
	local viewport = workspace.CurrentCamera and workspace.CurrentCamera.ViewportSize or Vector2.new(1280, 720)
	local minAxis = math.min(viewport.X, viewport.Y)
	local targetScale = 1.0

	if UIS.TouchEnabled then
		targetScale = (minAxis < 600) and 0.85 or 0.95
	else
		if viewport.X >= 1920 or viewport.Y >= 1080 then
			targetScale = 1.05
		elseif viewport.X <= 1366 then
			targetScale = 0.95
		end
	end
	UIScale.Scale = targetScale
end

-- Lucide Icon Dictionary (Cached & Self-Recoloring)
local IconsMap = {}
task.spawn(function()
	pcall(function()
		if game and game.HttpGet then
			local raw = game:HttpGet("https://raw.githubusercontent.com/KazeHub/UI/refs/heads/main/Icons.lua.txt")
			if raw then
				local func, err = loadstring(raw)
				local res = func and func()
				if type(res) == "table" then
					for k, v in pairs(res) do
						IconsMap[string.lower(tostring(k))] = tostring(v)
					end
				end
			end
		end
	end)
end)

local function FormatImage(id)
	if not id or id == "" then return "" end
	id = tostring(id)
	local lowerId = string.lower(id)
	if IconsMap[lowerId] then
		local mapped = IconsMap[lowerId]
		if not string.find(mapped, "rbxassetid://") and not string.find(mapped, "http") then
			local clean = string.gsub(mapped, "%D", "")
			if clean ~= "" then return "rbxassetid://" .. clean end
		end
		return mapped
	end
	if string.find(id, "rbxassetid://") or string.find(id, "http") then return id end
	local clean = string.gsub(id, "%D", "")
	if clean == "" then return id end
	return "rbxassetid://" .. clean
end

-- MODULE 1: CUSTOM PHYSICS-BASED ANIMATION ENGINE
-- =============================================================================
local AnimationEngine = {}
do
	local activeAnimations = {}

	-- Spring physics simulation solver
	function AnimationEngine:Spring(instance, property, target, damping, stiffness)
		damping = damping or 15
		stiffness = stiffness or 150

		if not activeAnimations[instance] then
			activeAnimations[instance] = {}
		end

		if activeAnimations[instance][property] then
			activeAnimations[instance][property].Conn:Disconnect()
		end

		local initialVal = instance[property]
		local currentVal = initialVal
		local velocity = (type(initialVal) == "number") and 0 
			or (typeof(initialVal) == "UDim2" and {0,0,0,0}) 
			or (typeof(initialVal) == "Color3" and {0,0,0}) 
			or 0

		local conn
		conn = RunService.Heartbeat:Connect(function(dt)
			dt = math.min(dt, 0.1) -- Cap delta-time to prevent physics explosion
			
			if typeof(target) == "number" then
				local force = (target - currentVal) * stiffness
				velocity = velocity + (force - velocity * damping) * dt
				currentVal = currentVal + velocity * dt

				if math.abs(target - currentVal) < 0.001 and math.abs(velocity) < 0.001 then
					instance[property] = target
					conn:Disconnect()
					activeAnimations[instance][property] = nil
				else
					instance[property] = currentVal
				end
			elseif typeof(target) == "UDim2" then
				local targetXSc, targetXOf = target.X.Scale, target.X.Offset
				local targetYSc, targetYOf = target.Y.Scale, target.Y.Offset
				
				local curXSc, curXOf = currentVal.X.Scale, currentVal.X.Offset
				local curYSc, curYOf = currentVal.Y.Scale, currentVal.Y.Offset

				local fXSc = (targetXSc - curXSc) * stiffness
				local fXOf = (targetXOf - curXOf) * stiffness
				local fYSc = (targetYSc - curYSc) * stiffness
				local fYOf = (targetYOf - curYOf) * stiffness

				velocity[1] = velocity[1] + (fXSc - velocity[1] * damping) * dt
				velocity[2] = velocity[2] + (fXOf - velocity[2] * damping) * dt
				velocity[3] = velocity[3] + (fYSc - velocity[3] * damping) * dt
				velocity[4] = velocity[4] + (fYOf - velocity[4] * damping) * dt

				curXSc = curXSc + velocity[1] * dt
				curXOf = curXOf + velocity[2] * dt
				curYSc = curYSc + velocity[3] * dt
				curYOf = curYOf + velocity[4] * dt

				if math.abs(targetXSc - curXSc) < 0.0001 and math.abs(targetXOf - curXOf) < 0.01 
				and math.abs(targetYSc - curYSc) < 0.0001 and math.abs(targetYOf - curYOf) < 0.01 then
					instance[property] = target
					conn:Disconnect()
					activeAnimations[instance][property] = nil
				else
					instance[property] = UDim2.new(curXSc, curXOf, curYSc, curYOf)
					currentVal = instance[property]
				end
			elseif typeof(target) == "Color3" then
				local rF = (target.R - currentVal.R) * stiffness
				local gF = (target.G - currentVal.G) * stiffness
				local bF = (target.B - currentVal.B) * stiffness

				velocity[1] = velocity[1] + (rF - velocity[1] * damping) * dt
				velocity[2] = velocity[2] + (gF - velocity[2] * damping) * dt
				velocity[3] = velocity[3] + (bF - velocity[3] * damping) * dt

				local curR = currentVal.R + velocity[1] * dt
				local curG = currentVal.G + velocity[2] * dt
				local curB = currentVal.B + velocity[3] * dt

				if math.abs(target.R - curR) < 0.001 and math.abs(target.G - curG) < 0.001 and math.abs(target.B - curB) < 0.001 then
					instance[property] = target
					conn:Disconnect()
					activeAnimations[instance][property] = nil
				else
					instance[property] = Color3.new(
						math.clamp(curR, 0, 1),
						math.clamp(curG, 0, 1),
						math.clamp(curB, 0, 1)
					)
					currentVal = instance[property]
				end
			else
				-- Fallback to simple tween for unmapped types
				instance[property] = target
				conn:Disconnect()
			end
		end)

		activeAnimations[instance][property] = {Conn = conn}
	end

	function AnimationEngine:Clean(instance)
		if activeAnimations[instance] then
			for _, anim in pairs(activeAnimations[instance]) do
				anim.Conn:Disconnect()
			end
			activeAnimations[instance] = nil
		end
	end
end

function ThemeEngine:RegisterText(instance, isMuted)
	if not instance then return end
	local registry = isMuted and self.RegisteredMutedText or self.RegisteredText
	registry[instance] = true
	
	instance.TextColor3 = isMuted and KazeUI.CurrentTheme.MutedText or KazeUI.CurrentTheme.Text
	if instance:IsA("TextBox") then
		instance.PlaceholderColor3 = KazeUI.CurrentTheme.MutedText
	end

	instance.Destroying:Connect(function()
		registry[instance] = nil
	end)
end

function ThemeEngine:RegisterBorder(instance)
	if not instance then return end
	self.RegisteredBorders[instance] = true
	instance.Color = KazeUI.CurrentTheme.Border

	instance.Destroying:Connect(function()
		self.RegisteredBorders[instance] = nil
	end)
end

function ThemeEngine:OnThemeChanged(instance, callback)
	if not instance then return end
	self.ThemeCallbacks[instance] = callback
	task.spawn(callback, KazeUI.CurrentTheme)

	instance.Destroying:Connect(function()
		self.ThemeCallbacks[instance] = nil
	end)
end

function ThemeEngine:RegisterGlow(instance, callback)
	if not instance then return end
	self.GlowCallbacks[instance] = callback
	task.spawn(callback, KazeUI.GlowColor)

	instance.Destroying:Connect(function()
		self.GlowCallbacks[instance] = nil
	end)
end

function ThemeEngine:ApplyTheme(themeName)
	local theme = KazeUI.Themes[themeName] or KazeUI.Themes.Obsidian
	KazeUI.CurrentTheme = theme
	
	-- Refresh all elements via physics-based animation
	for textInst in pairs(self.RegisteredText) do
		AnimationEngine:Spring(textInst, "TextColor3", theme.Text, 18, 160)
	end

	for mutedInst in pairs(self.RegisteredMutedText) do
		AnimationEngine:Spring(mutedInst, "TextColor3", theme.MutedText, 18, 160)
		if mutedInst:IsA("TextBox") then
			AnimationEngine:Spring(mutedInst, "PlaceholderColor3", theme.MutedText, 18, 160)
		end
	end

	for borderInst in pairs(self.RegisteredBorders) do
		AnimationEngine:Spring(borderInst, "Color", theme.Border, 18, 160)
	end

	for inst, callback in pairs(self.ThemeCallbacks) do
		task.spawn(callback, theme)
	end

	local imageId = theme.BackgroundImage or ""
	local imgTrans = theme.BackgroundImageTransparency or 1
	local overlayTrans = theme.OverlayTransparency or 0.25
	local validImage = (imageId ~= "") and (imgTrans < 1)

	for _, wp in ipairs(self.WallpaperRegistrations) do
		if wp.ImageLabel and wp.ImageLabel.Parent then
			if validImage then
				wp.ImageLabel.Image = FormatImage(imageId)
				AnimationEngine:Spring(wp.ImageLabel, "ImageTransparency", imgTrans, 14, 100)
				AnimationEngine:Spring(wp.OverlayFrame, "BackgroundTransparency", overlayTrans, 14, 100)
			else
				AnimationEngine:Spring(wp.ImageLabel, "ImageTransparency", 1, 14, 100)
				AnimationEngine:Spring(wp.OverlayFrame, "BackgroundTransparency", 1, 14, 100)
			end
		end
	end
end

-- Ambient Glowing Breather Effect
local GlowLoop = nil
local function StartGlowOscillator()
	if GlowLoop then GlowLoop:Disconnect() end
	local elapsed = 0
	GlowLoop = RunService.Heartbeat:Connect(function(dt)
		elapsed = elapsed + dt
		local mult = 0.9 + math.sin(elapsed * 2.5) * 0.1
		for instance, prop in pairs(ThemeEngine.GlowRegistrations) do
			if instance and instance.Parent then
				local color = KazeUI.GlowColor
				local h, s, v = Color3.toHSV(color)
				instance[prop] = Color3.fromHSV(h, s * 0.9, math.clamp(v * mult, 0, 1))
			else
				ThemeEngine.GlowRegistrations[instance] = nil
			end
		end
	end)
end

function ThemeEngine:RegisterGlowAnimator(instance, property)
	self.GlowRegistrations[instance] = property or "BackgroundColor3"
	if not GlowLoop then StartGlowOscillator() end

	instance.Destroying:Connect(function()
		self.GlowRegistrations[instance] = nil
	end)
end

		iconLabel.BackgroundTransparency = 1

	titleLabel.Position = UDim2.fromOffset(leftPadding, 12)

	TweenService:Create(bar, TweenInfo.new(duration, Enum.EasingStyle.Linear), {Size = UDim2.new(0, 0, 1, 0)}):Play()
	task.delay(duration, function()
		if notifHolder and notifHolder.Parent then dismiss() end
	end)

	if state == nil then state = not self.Host.Overlay.Visible end
	self.Host.Overlay.Visible = state
	
	if state then
		self.Host.TextBox.Text = ""
		self:RenderSearch("")
		self.Host.TextBox:CaptureFocus()
		AnimationEngine:Spring(self.Host.Container, "Position", UDim2.fromScale(0.5, 0.5), 14, 110)
		TweenService:Create(self.Host.Overlay, TweenInfo.new(0.2), {BackgroundTransparency = 0.5}):Play()
	else
		self.Host.TextBox:ReleaseFocus()
		AnimationEngine:Spring(self.Host.Container, "Position", UDim2.fromScale(0.5, 0.4), 14, 110)
		TweenService:Create(self.Host.Overlay, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()
	end
end

	end

	query = string.lower(query)
	local matches = 0

		if query == "" or string.find(string.lower(item.Name), query, 1, true) or string.find(string.lower(item.Category), query, 1, true) then

local function MakeWindowDraggable(dragPart, targetPart)
	local dragging, dragStart, startPos, dragInput
	
	dragPart.InputBegan:Connect(function(input)
		if KazeUI.IsResizing then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = targetPart.Position
			LockCamera(true)

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
					LockCamera(false)
				end
			end)
		end
	end)

	dragPart.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)

	UIS.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			local delta = input.Position - dragStart
			local scale = UIScale.Scale
			targetPart.Position = UDim2.new(
				startPos.X.Scale, startPos.X.Offset + (delta.X / scale),
				startPos.Y.Scale, startPos.Y.Offset + (delta.Y / scale)
			)
		end
	end)
end

	NotificationManager:Notify(config)
end

function KazeUI:SaveConfig(name)
	if not name or name == "" then return end
	local data = {}
	for flag, val in pairs(self.Flags) do
		data[flag] = val.Get()
	end
	if writefile then
		local ok, encoded = pcall(HttpService.JSONEncode, HttpService, data)
		if ok then
			local folder = self.AuthorName
			pcall(function()
				if makefolder then makefolder(folder) end
			end)
			writefile(folder .. "/" .. name .. ".json", encoded)
			self:Notify({Title = "Config Saved", Content = "Saved visual state to " .. name, Duration = 2})
		end
	end
end

function KazeUI:LoadConfig(name)
	if readfile then
		local folder = self.AuthorName
		local success, content = pcall(readfile, folder .. "/" .. name .. ".json")
		if success then
			local ok, data = pcall(HttpService.JSONDecode, HttpService, content)
			if ok and type(data) == "table" then
				for flag, val in pairs(data) do
					if self.Flags[flag] then
						self.Flags[flag].Set(val)
					end
				end
				self:Notify({Title = "Config Loaded", Content = "Applied loaded state from " .. name, Duration = 2})
			end
		end
	end
end

-- Shared Component Instantiators
local function ToggleScrollRecursion(element, state)
	local cur = element.Parent
	while cur do
		if cur:IsA("ScrollingFrame") then cur.ScrollingEnabled = state end
		cur = cur.Parent
	end
end

local function CreateSliderInput(parentFrame, title, min, max, default, callback, flag)
	local val = math.clamp(default or min, min, max)

	local wrapper = Instance.new("Frame")
	wrapper.Size = UDim2.new(1, 0, 0, 0)
	wrapper.BackgroundTransparency = 1
	wrapper.AutomaticSize = Enum.AutomaticSize.Y
	wrapper.Parent = parentFrame

	local card = Instance.new("Frame", wrapper)
	card.Size = UDim2.new(0.95, 0, 0, 0)
	card.AnchorPoint = Vector2.new(0.5, 0)
	card.Position = UDim2.new(0.5, 0, 0, 3)
	card.AutomaticSize = Enum.AutomaticSize.Y
	card.BackgroundColor3 = KazeUI.Themes.Obsidian.Background
	card.BackgroundTransparency = 0.1
	Instance.new("UICorner", card).CornerRadius = UDim.new(0, 14)

	local border = Instance.new("UIStroke", card)
	border.Thickness = 1.2
	ThemeEngine:RegisterBorder(border)

	local layout = Instance.new("UIListLayout", card)
	layout.Padding = UDim.new(0, 6)
	layout.HorizontalAlignment = Enum.HorizontalAlignment.Center

	local pad = Instance.new("UIPadding", card)
	pad.PaddingTop = UDim.new(0, 10)
	pad.PaddingBottom = UDim.new(0, 10)
	pad.PaddingLeft = UDim.new(0, 12)
	pad.PaddingRight = UDim.new(0, 12)

	local header = Instance.new("Frame", card)
	header.Size = UDim2.new(1, 0, 0, 20)
	header.BackgroundTransparency = 1

	local tLabel = Instance.new("TextLabel", header)
	tLabel.Size = UDim2.new(0.7, 0, 1, 0)
	tLabel.BackgroundTransparency = 1
	tLabel.Text = title
	tLabel.Font = Enum.Font.GothamBold
	tLabel.TextSize = 13
	tLabel.TextXAlignment = Enum.TextXAlignment.Left
	ThemeEngine:RegisterText(tLabel, false)

	local valLabel = Instance.new("TextLabel", header)
	valLabel.Size = UDim2.new(0.3, 0, 1, 0)
	valLabel.Position = UDim2.fromScale(0.7, 0)
	valLabel.BackgroundTransparency = 1
	valLabel.Text = tostring(val)
	valLabel.Font = Enum.Font.GothamBold
	valLabel.TextSize = 12
	valLabel.TextXAlignment = Enum.TextXAlignment.Right
	ThemeEngine:RegisterText(valLabel, false)

	local sliderTrack = Instance.new("Frame", card)
	sliderTrack.Size = UDim2.new(1, 0, 0, 6)
	sliderTrack.BackgroundColor3 = Color3.fromRGB(30, 30, 40)
	Instance.new("UICorner", sliderTrack).CornerRadius = UDim.new(1, 0)

	local sliderFill = Instance.new("Frame", sliderTrack)
	sliderFill.Size = UDim2.fromScale(0, 1)
	sliderFill.BackgroundColor3 = KazeUI.GlowColor
	Instance.new("UICorner", sliderFill).CornerRadius = UDim.new(1, 0)
	ThemeEngine:RegisterGlowAnimator(sliderFill, "BackgroundColor3")

	local knob = Instance.new("Frame", sliderTrack)
	knob.Size = UDim2.fromOffset(12, 12)
	knob.AnchorPoint = Vector2.new(0.5, 0.5)
	knob.Position = UDim2.fromScale(0, 0.5)
	knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

	local activeDrag = false

	local function updateSlider(input)
		local x = input.Position.X - sliderTrack.AbsolutePosition.X
		local pct = math.clamp(x / sliderTrack.AbsoluteSize.X, 0, 1)
		val = min + (max - min) * pct
		val = math.clamp(math.floor(val + 0.5), min, max)

		valLabel.Text = tostring(val)
		local visualPct = (val - min) / (max - min)
		
		AnimationEngine:Spring(sliderFill, "Size", UDim2.fromScale(visualPct, 1), 14, 150)
		AnimationEngine:Spring(knob, "Position", UDim2.fromScale(visualPct, 0.5), 14, 150)

		if callback then task.spawn(callback, val) end
	end

	-- Initial scaling
	task.spawn(function()
		task.wait(0.05)
		local visualPct = (val - min) / (max - min)
		sliderFill.Size = UDim2.fromScale(visualPct, 1)
		knob.Position = UDim2.fromScale(visualPct, 0.5)
	end)

	sliderTrack.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			activeDrag = true
			LockCamera(true)
			ToggleScrollRecursion(sliderTrack, false)
			updateSlider(input)

			local connMove, connEnd
			connMove = UIS.InputChanged:Connect(function(moveInput)
				if activeDrag and (moveInput.UserInputType == Enum.UserInputType.MouseMovement or moveInput.UserInputType == Enum.UserInputType.Touch) then
					updateSlider(moveInput)
				end
			end)
			connEnd = UIS.InputEnded:Connect(function(endInput)
				if endInput.UserInputType == Enum.UserInputType.MouseButton1 or endInput.UserInputType == Enum.UserInputType.Touch then
					activeDrag = false
					LockCamera(false)
					ToggleScrollRecursion(sliderTrack, true)
					connMove:Disconnect()
					connEnd:Disconnect()
				end
			end)
		end
	end)

	if flag then
		KazeUI.Flags[flag] = {
			Get = function() return val end,
			Set = function(v)
				val = math.clamp(v, min, max)
				valLabel.Text = tostring(val)
				local pct = (val - min) / (max - min)
				AnimationEngine:Spring(sliderFill, "Size", UDim2.fromScale(pct, 1), 14, 150)
				AnimationEngine:Spring(knob, "Position", UDim2.fromScale(pct, 0.5), 14, 150)
			end
		}
	end

	return {}
end

local function AttachElementsToAPI(apiTable, parentFrame)
	function apiTable:AddLabel(text)
		local l = Instance.new("TextLabel", parentFrame)
		l.Size = UDim2.new(1, -12, 0, 22)
		l.BackgroundTransparency = 1
		l.Text = tostring(text or "")
		l.Font = Enum.Font.GothamMedium
		l.TextSize = 13
		l.TextXAlignment = Enum.TextXAlignment.Left
		ThemeEngine:RegisterText(l, false)
		return l
	end

	function apiTable:Paragraph(arg1, arg2)
		local title = type(arg1) == "table" and arg1.Title or arg1
		local content = type(arg1) == "table" and (arg1.Content or arg1.Description) or arg2
		return CreateParagraph(parentFrame, title, content)
	end

	function apiTable:Button(arg1, arg2, arg3)
		local title = type(arg1) == "table" and arg1.Title or arg1
		local desc = type(arg1) == "table" and (arg1.Description or arg1.Content or "") or arg2
		local callback = type(arg1) == "table" and arg1.Callback or arg3
		return CreateClickableParagraph(parentFrame, title, desc, callback)
	end

	function apiTable:Toggle(arg1, arg2, arg3, arg4, arg5)
		local title, desc, def, callback, flag = "", "", false, nil, nil
		if type(arg1) == "table" then
			title, desc, def, callback, flag = arg1.Title, arg1.Description or arg1.Content or "", arg1.Default, arg1.Callback, arg1.Flag
		else
			title, desc, def, callback, flag = arg1, arg2, arg3, arg4, arg5
		end
		return CreateToggleSwitch(parentFrame, title, desc, def, callback, flag)
	end

	function apiTable:TextBox(arg1, arg2, arg3, arg4, arg5)
		local title, place, def, callback, flag = "", "", "", nil, nil
		if type(arg1) == "table" then
			title, place, def, callback, flag = arg1.Title, arg1.Placeholder, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, place, def, callback, flag = arg1, arg2, arg3, arg4, arg5
		end
		return CreateTextBoxInput(parentFrame, title, place, def, callback, flag)
	end

	function apiTable:Slider(arg1, arg2, arg3, arg4, arg5, arg6)
		local title, min, max, def, callback, flag = "", 0, 100, 50, nil, nil
		if type(arg1) == "table" then
			title, min, max, def, callback, flag = arg1.Title, arg1.Min, arg1.Max, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, min, max, def, callback, flag = arg1, arg2, arg3, arg4, arg5, arg6
		end
		return CreateSliderInput(parentFrame, title, min, max, def, callback, flag)
	end

	function apiTable:Dropdown(arg1, arg2, arg3, arg4, arg5, arg6)
		local title, vals, def, callback, multi, flag = "", {}, nil, nil, false, nil
		if type(arg1) == "table" then
			title, vals, def, multi, callback, flag = arg1.Title, arg1.Values or arg1.Options, arg1.Default, arg1.Multi or false, arg1.Callback, arg1.Flag
		else
			title, vals, def, callback, multi, flag = arg1, arg2, arg3, arg4, arg5 or false, arg6
		end
		return CreateDropdown(parentFrame, title, vals, def, callback, multi, flag)
	end

	function apiTable:ColorPicker(arg1, arg2, arg3, arg4)
		local title, def, cb, flag = "", Color3.fromRGB(255, 255, 255), nil, nil
		if type(arg1) == "table" then
			title, def, cb, flag = arg1.Title, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, def, cb, flag = arg1, arg2, arg3, arg4
		end
		return CreateColorPicker(parentFrame, title, def, cb, flag)
	end

	function apiTable:Keybind(arg1, arg2, arg3, arg4)
		local title, def, cb, flag = "", Enum.KeyCode.Unknown, nil, nil
		if type(arg1) == "table" then
			title, def, cb, flag = arg1.Title, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, def, cb, flag = arg1, arg2, arg3, arg4
		end
		return CreateKeybindInput(parentFrame, title, def, cb, flag)
	end

	function apiTable:SectionUI(title)
		local sec = {
			Frame = Instance.new("Frame", parentFrame)
		}
		sec.Frame.Size = UDim2.new(0.95, 0, 0, 0)
		sec.Frame.AutomaticSize = Enum.AutomaticSize.Y
		sec.Frame.BackgroundTransparency = 1

		local l = Instance.new("UIListLayout", sec.Frame)
		l.Padding = UDim.new(0, 6)
		l.HorizontalAlignment = Enum.HorizontalAlignment.Center

		CreateVisualSectionDivider(sec.Frame, title)
		AttachElementsToAPI(sec, sec.Frame)
		return sec
	end

	function apiTable:Section(title)
		return CreateVisualSectionDivider(parentFrame, title)
	end
end

function KazeUI:CreateWindow(config)
	config = config or {}
	local title = config.Title or "Kaze UI v4"
	local author = config.Author or "Unknown"
	local version = config.Version or "4.0.0"
	local mainIcon = FormatImage(config.Icon or "house")

	self.AuthorName = author

	local windowFrame = Instance.new("Frame")
	windowFrame.Size = UDim2.fromOffset(620, 420)
	windowFrame.Position = UDim2.fromScale(0.5, 0.5)
	windowFrame.AnchorPoint = Vector2.new(0.5, 0.5)
	windowFrame.BackgroundColor3 = self.Themes.Obsidian.Background
	windowFrame.BorderSizePixel = 0
	windowFrame.Parent = ScreenGui
	Instance.new("UICorner", windowFrame).CornerRadius = UDim.new(0, 24)

	local winStroke = Instance.new("UIStroke", windowFrame)
	winStroke.Thickness = 1.5
	ThemeEngine:RegisterBorder(winStroke)

	-- Advanced shadow
	local shadow = Instance.new("Frame", windowFrame)
	shadow.Size = UDim2.new(1, 16, 1, 16)
	shadow.Position = UDim2.fromOffset(-8, -8)
	shadow.BackgroundTransparency = 1
	shadow.ZIndex = -1
	Instance.new("UICorner", shadow).CornerRadius = UDim.new(0, 28)
	local shadowStroke = Instance.new("UIStroke", shadow)
	shadowStroke.Thickness = 6
	shadowStroke.Color = Color3.fromRGB(0, 0, 0)
	shadowStroke.Transparency = 0.8

	local contentClipper = Instance.new("CanvasGroup", windowFrame)
	contentClipper.Size = UDim2.fromScale(1, 1)
	contentClipper.BackgroundTransparency = 1
	contentClipper.BorderSizePixel = 0
	Instance.new("UICorner", contentClipper).CornerRadius = UDim.new(0, 24)

	-- Wallpaper overlay implementation
	local wallpaper = Instance.new("ImageLabel", contentClipper)
	wallpaper.Size = UDim2.fromScale(1, 1)
	wallpaper.BackgroundTransparency = 1
	wallpaper.ScaleType = Enum.ScaleType.Crop
	wallpaper.ZIndex = -2

	local overlay = Instance.new("Frame", contentClipper)
	overlay.Size = UDim2.fromScale(1, 1)
	overlay.BackgroundColor3 = Color3.fromRGB(0,0,0)
	overlay.ZIndex = -1

	table.insert(ThemeEngine.WallpaperRegistrations, {
		ImageLabel = wallpaper,
		OverlayFrame = overlay,
	})

	local topBar = Instance.new("Frame", contentClipper)
	topBar.Size = UDim2.new(1, 0, 0, 52)
	topBar.BackgroundTransparency = 1
	MakeWindowDraggable(topBar, windowFrame)

	local avatar = Instance.new("ImageLabel", topBar)
	avatar.Size = UDim2.fromOffset(32, 32)
	avatar.Position = UDim2.fromOffset(14, 10)
	avatar.Image = mainIcon
	avatar.BackgroundTransparency = 1
	Instance.new("UICorner", avatar).CornerRadius = UDim.new(1, 0)

	local titleLabel = Instance.new("TextLabel", topBar)
	titleLabel.Size = UDim2.new(0.5, 0, 0, 20)
	titleLabel.Position = UDim2.fromOffset(56, 8)
	titleLabel.BackgroundTransparency = 1
	titleLabel.Text = title
	titleLabel.Font = Enum.Font.GothamBold
	titleLabel.TextSize = 13
	titleLabel.TextXAlignment = Enum.TextXAlignment.Left
	ThemeEngine:RegisterText(titleLabel, false)

	local authorLabel = Instance.new("TextLabel", topBar)
	authorLabel.Size = UDim2.new(0.5, 0, 0, 16)
	authorLabel.Position = UDim2.fromOffset(56, 26)
	authorLabel.BackgroundTransparency = 1
	authorLabel.Text = author .. " • " .. version
	authorLabel.Font = Enum.Font.GothamMedium
	authorLabel.TextSize = 11
	authorLabel.TextXAlignment = Enum.TextXAlignment.Left
	ThemeEngine:RegisterText(authorLabel, true)

	-- Floating Control Center Button (Command Palette quick trigger)
	local cmdBtn = Instance.new("ImageButton", topBar)
	cmdBtn.Size = UDim2.fromOffset(24, 24)
	cmdBtn.Position = UDim2.new(1, -120, 0.5, 0)
	cmdBtn.AnchorPoint = Vector2.new(1, 0.5)
	cmdBtn.BackgroundTransparency = 1
	cmdBtn.Image = "rbxassetid://101768155599700" -- Standard UI Search Node
	ThemeEngine:OnThemeChanged(cmdBtn, function(t) cmdBtn.ImageColor3 = t.MutedText end)
	cmdBtn.MouseButton1Click:Connect(function()
		CommandPalette:Toggle(true)
	end)

	local function createMinMaxBtn(offsetX, color, action)
		local b = Instance.new("TextButton", topBar)
		b.Size = UDim2.fromOffset(12, 12)
		b.Position = UDim2.new(1, offsetX, 0.5, 0)
		b.AnchorPoint = Vector2.new(1, 0.5)
		b.BackgroundColor3 = color
		b.BorderSizePixel = 0
		b.Text = ""
		Instance.new("UICorner", b).CornerRadius = UDim.new(1, 0)
		b.MouseButton1Click:Connect(action)
		return b
	end

	-- Window control loops
	local minimized = false
	local origSize = windowFrame.Size
	local origPos = windowFrame.Position

	local function toggleMinimize()
		minimized = not minimized
		if minimized then
			origSize = windowFrame.Size
			origPos = windowFrame.Position
			AnimationEngine:Spring(windowFrame, "Size", UDim2.fromOffset(620, 52), 15, 120)
		else
			AnimationEngine:Spring(windowFrame, "Size", origSize, 15, 120)
		end
	end

	local function destroyWindow()
		AnimationEngine:Clean(windowFrame)
		windowFrame:Destroy()
	end

	createMinMaxBtn(-20, Color3.fromRGB(239, 68, 68), destroyWindow)
	createMinMaxBtn(-44, Color3.fromRGB(234, 179, 8), toggleMinimize)

	local splitFrame = Instance.new("Frame", contentClipper)
	splitFrame.Size = UDim2.new(1, 0, 1, -52)
	splitFrame.Position = UDim2.fromOffset(0, 52)
	splitFrame.BackgroundTransparency = 1

	local sidebar = Instance.new("Frame", splitFrame)
	sidebar.Size = UDim2.new(0, 180, 1, 0)
	sidebar.BackgroundTransparency = 1

	local sideDivider = Instance.new("Frame", splitFrame)
	sideDivider.Size = UDim2.new(0, 1, 1, 0)
	sideDivider.Position = UDim2.fromOffset(180, 0)
	sideDivider.BorderSizePixel = 0
	ThemeEngine:OnThemeChanged(sideDivider, function(t) sideDivider.BackgroundColor3 = t.Border end)

	local pagesHost = Instance.new("Frame", splitFrame)
	pagesHost.Size = UDim2.new(1, -181, 1, 0)
	pagesHost.Position = UDim2.fromOffset(181, 0)
	pagesHost.BackgroundTransparency = 1

	local sidebarScroller = Instance.new("ScrollingFrame", sidebar)
	sidebarScroller.Size = UDim2.new(1, -16, 1, -16)
	sidebarScroller.Position = UDim2.fromOffset(8, 8)
	sidebarScroller.BackgroundTransparency = 1
	sidebarScroller.BorderSizePixel = 0
	sidebarScroller.ScrollBarThickness = 2
	sidebarScroller.AutomaticCanvasSize = Enum.AutomaticSize.Y

	local sidebarLayout = Instance.new("UIListLayout", sidebarScroller)
	sidebarLayout.Padding = UDim.new(0, 6)

	-- Resizing handle support
	local resizeHandle = Instance.new("ImageButton", windowFrame)
	resizeHandle.Size = UDim2.fromOffset(16, 16)
	resizeHandle.Position = UDim2.new(1, -16, 1, -16)
	resizeHandle.BackgroundTransparency = 1
	resizeHandle.Image = "rbxassetid://10747373117"
	ThemeEngine:OnThemeChanged(resizeHandle, function(t) resizeHandle.ImageColor3 = t.MutedText end)

	resizeHandle.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			self.IsResizing = true
			LockCamera(true)
			local startSize = windowFrame.AbsoluteSize
			local startPos = input.Position

			local connMove, connEnd
			connMove = UIS.InputChanged:Connect(function(moveInput)
				if moveInput.UserInputType == Enum.UserInputType.MouseMovement or moveInput.UserInputType == Enum.UserInputType.Touch then
					local delta = moveInput.Position - startPos
					local scale = UIScale.Scale
					windowFrame.Size = UDim2.fromOffset(
						math.clamp(startSize.X + (delta.X / scale), 400, 900),
						math.clamp(startSize.Y + (delta.Y / scale), 300, 700)
					)
				end
			end)
			connEnd = UIS.InputEnded:Connect(function(endInput)
				if endInput.UserInputType == Enum.UserInputType.MouseButton1 or endInput.UserInputType == Enum.UserInputType.Touch then
					self.IsResizing = false
					LockCamera(false)
					connMove:Disconnect()
					connEnd:Disconnect()
				end
			end)
		end
	end)

	local Win = {
		_Pages = {},
		_Tabs = {},
	}

	function Win:CreateTab(arg1, arg2)
		local name = tostring(type(arg1) == "table" and (arg1.Title or arg1.Name) or arg1 or "Tab")
		local iconId = type(arg1) == "table" and arg1.Icon or arg2

		local tabBtn = Instance.new("TextButton", sidebarScroller)
		tabBtn.Size = UDim2.new(1, 0, 0, 36)
		tabBtn.BackgroundColor3 = Color3.fromRGB(24, 24, 30)
		tabBtn.Text = ""
		Instance.new("UICorner", tabBtn).CornerRadius = UDim.new(0, 10)

		local tabStroke = Instance.new("UIStroke", tabBtn)
		tabStroke.Thickness = 1
		ThemeEngine:RegisterBorder(tabStroke)

		local tLabel = Instance.new("TextLabel", tabBtn)
		tLabel.Size = UDim2.new(1, -32, 1, 0)
		tLabel.Position = UDim2.fromOffset(12, 0)
		tLabel.BackgroundTransparency = 1
		tLabel.Text = name
		tLabel.Font = Enum.Font.GothamMedium
		tLabel.TextSize = 12
		tLabel.TextXAlignment = Enum.TextXAlignment.Left
		ThemeEngine:RegisterText(tLabel, false)

		local pageFrame = Instance.new("ScrollingFrame", pagesHost)
		pageFrame.Size = UDim2.fromScale(1, 1)
		pageFrame.BackgroundTransparency = 1
		pageFrame.BorderSizePixel = 0
		pageFrame.ScrollBarThickness = 3
		pageFrame.Visible = false
		pageFrame.AutomaticCanvasSize = Enum.AutomaticSize.Y

		local pageLayout = Instance.new("UIListLayout", pageFrame)
		pageLayout.Padding = UDim.new(0, 6)
		pageLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center

		local pagePad = Instance.new("UIPadding", pageFrame)
		pagePad.PaddingTop = UDim.new(0, 10)
		pagePad.PaddingBottom = UDim.new(0, 10)

		local function selectTab()
			for _, t in ipairs(Win._Tabs) do
				t.Page.Visible = false
				AnimationEngine:Spring(t.Button, "BackgroundColor3", Color3.fromRGB(24, 24, 30), 15, 120)
			end
			pageFrame.Visible = true
			AnimationEngine:Spring(tabBtn, "BackgroundColor3", Color3.fromRGB(40, 40, 50), 15, 120)
		end

		tabBtn.MouseButton1Click:Connect(selectTab)

		table.insert(Win._Tabs, {Button = tabBtn, Page = pageFrame})
		if #Win._Tabs == 1 then selectTab() end

		local tabAPI = {}
		AttachElementsToAPI(tabAPI, pageFrame)

		CommandPalette:Register("Go to Tab: " .. name, "Navigation", function()
			selectTab()
		end)

		return tabAPI
	end

	-- Live Setup API Wrappers
	Win.CreateConfigManager = function(tab)
		local sec = tab:SectionUI("Configs")
		local cName = "Default"
		sec:TextBox("Config Name", "Type custom name...", "Default", function(v) cName = v end)
		sec:Button("Save Profile", "Serialize current elements", function() KazeUI:SaveConfig(cName) end)
		sec:Button("Load Profile", "Apply targeted elements", function() KazeUI:LoadConfig(cName) end)
	end

	Win.CreateThemeManager = function(tab)
		local sec = tab:SectionUI("Personalization")
		sec:Dropdown("UI Preset Theme", {"Obsidian", "CustomWallpaper"}, "Obsidian", function(val)
			ThemeEngine:ApplyTheme(val)
		end)
		sec:ColorPicker("Accent Color", KazeUI.GlowColor, function(c)
			KazeUI.GlowColor = c
		end)
	end

	ThemeEngine:ApplyTheme(config.Theme or "Obsidian")
	
	-- Fire launch transition
	windowFrame.Size = UDim2.fromOffset(100, 100)
	windowFrame.BackgroundTransparency = 1
	AnimationEngine:Spring(windowFrame, "Size", UDim2.fromOffset(620, 420), 12, 100)

	return Win
end

return KazeUI
