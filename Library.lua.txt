-- Adaptive Scaling Engine
local UIScale = Instance.new("UIScale")
UIScale.Parent = ScreenGui

local function UpdateScale()
	local size = workspace.CurrentCamera.ViewportSize
	local width, height = size.X, size.Y
	local minAxis = math.min(width, height)
	local targetScale = 1.0

	if UIS.TouchEnabled then
		targetScale = (minAxis < 500) and 0.85 or 1.0
	else
		if width >= 1920 or height >= 1080 then
			targetScale = 1.12
		elseif width <= 1366 then
			targetScale = 0.90
		else
			targetScale = 1.0
		end
	end
	UIScale.Scale = targetScale
	ScreenGui.IgnoreGuiInset = true
end

UpdateScale()
workspace.CurrentCamera:GetPropertyChangedSignal("ViewportSize"):Connect(UpdateScale)

-- Lucide Icons Setup
local IconsMap = {}
pcall(function()
	if game and game.HttpGet then
		local rawIcons = game:HttpGet("https://raw.githubusercontent.com/KazeHub/UI/refs/heads/main/Icons.lua.txt")
		if rawIcons then
			local s, res = pcall(function()
				local func = loadstring(rawIcons)
				return func and func()
			end)
			if type(res) == "table" then
				for k, v in pairs(res) do IconsMap[string.lower(tostring(k))] = tostring(v) end
			end
		end
	end
end)

local function FormatImage(id)
	if not id or id == "" then return "" end
	id = tostring(id)
	local lowerId = string.lower(id)
	if IconsMap[lowerId] then
		local mapped = IconsMap[lowerId]
		if not string.find(mapped, "rbxassetid://") and not string.find(mapped, "http") then
			local clean = string.gsub(mapped, "%D", "")
			if clean ~= "" then return "rbxassetid://" .. clean end
		end
		return mapped
	end
	if string.find(id, "rbxassetid://") or string.find(id, "http") then return id end
	local clean = string.gsub(id, "%D", "")
	if clean == "" then return id end
	return "rbxassetid://" .. clean
end

-- Structural drag engine
local lastNormalPosition = UDim2.fromScale(0.5, 0.5)
local lastNormalAnchor = Vector2.new(0.5, 0.5)

local function MakeDraggable(dragPart, targetPart)
	local dragging = false
	local dragInput, dragStart, startPos
	local dragActive = false
	dragPart.Active = true
	
	local function update(input)
		if not dragStart or not startPos then return end
		local delta = input.Position - dragStart
		local scale = UIScale.Scale
		if dragActive or delta.Magnitude > 6 then
			dragActive = true
			targetPart.Position = UDim2.new(
				startPos.X.Scale, startPos.X.Offset + (delta.X / scale),
				startPos.Y.Scale, startPos.Y.Offset + (delta.Y / scale)
			)
			if targetPart.Name == "Frame" or targetPart:IsA("Frame") then
				lastNormalPosition = targetPart.Position
				lastNormalAnchor = targetPart.AnchorPoint
			end
		end
	end
	
	dragPart.InputBegan:Connect(function(input)
		if KazeUI.IsResizing then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragActive = false
			dragStart = input.Position
			startPos = targetPart.Position
			LockCamera()
			local changeCon
			changeCon = input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
					dragInput = nil
					UnlockCamera()
					if changeCon then changeCon:Disconnect() end
				end
			end)
		end
	end)
	
	dragPart.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	
	UIS.InputChanged:Connect(function(input)
		if input == dragInput and dragging then update(input) end
	end)
end

-- Robotic Panel Constructors
function KazeUI.GetColor(elevation)
	local theme = KazeUI.CurrentTheme
	local h, s, v = Color3.toHSV(theme.Background)
	local factor = theme.LightnessFactor or 1
	local offset = elevation * factor
	return Color3.fromHSV(h, s, math.clamp(v + offset, 0, 1))
end

function KazeUI:AddPanel(inst, elevation, ignoreColor, ignoreTrans)
	table.insert(KazeUI.PanelElements, {Inst = inst, Elevation = elevation or 0, IgnoreColor = ignoreColor, IgnoreTransparency = ignoreTrans})
	if inst and inst:IsA("GuiObject") then
		if not ignoreColor then inst.BackgroundColor3 = KazeUI.GetColor(elevation or 0) end
		if not ignoreTrans then
			local baseTrans = KazeUI.BackgroundTransparency or 0.12
			inst.BackgroundTransparency = math.clamp(baseTrans - ((elevation or 0) * 0.08), 0.01, 1)
		end
	end
end

function KazeUI:SetBackgroundColor(color)
	for _, data in ipairs(KazeUI.PanelElements) do
		if data.Inst and data.Inst.Parent and data.Inst:IsA("GuiObject") and not data.IgnoreColor then
			data.Inst.BackgroundColor3 = KazeUI.GetColor(data.Elevation)
		end
	end
end

function KazeUI:SetTransparency(alpha)
	KazeUI.BackgroundTransparency = alpha
	for _, data in ipairs(KazeUI.PanelElements) do
		local inst = data.Inst
		if inst and inst.Parent and inst:IsA("GuiObject") and not data.IgnoreTransparency then
			inst.BackgroundTransparency = math.clamp(alpha - ((data.Elevation or 0) * 0.08), 0.01, 1)
		end
	end
end

function KazeUI:SetTheme(themeName)
	local targetTheme = KazeUI.Themes[themeName] or KazeUI.Themes.Obsidian
	KazeUI.CurrentTheme = targetTheme
	KazeUI.BackgroundColor = targetTheme.Background
	KazeUI:SetBackgroundColor(targetTheme.Background)
	KazeUI:SetTransparency(KazeUI.BackgroundTransparency or 0.12)
	
	for _, inst in ipairs(KazeUI.TextElements) do
		if inst and inst.Parent then inst.TextColor3 = targetTheme.Text end
	end
	for _, inst in ipairs(KazeUI.MutedTextElements) do
		if inst and inst.Parent then inst.TextColor3 = targetTheme.MutedText end
	end
	for _, inst in ipairs(KazeUI.BorderElements) do
		if inst and inst.Parent then inst.Color = targetTheme.Border end
	end

	local bgImgId = targetTheme.BackgroundImage or ""
	local bgImgTrans = targetTheme.BackgroundImageTransparency or 1
	local bgOverlayTrans = targetTheme.OverlayTransparency or 0.25
	local hasImage = (bgImgId ~= "") and (bgImgTrans < 1)

	for _, wp in ipairs(KazeUI.WallpaperElements) do
		if wp.ImageLabel and wp.ImageLabel.Parent then
			if hasImage then
				wp.ImageLabel.Image = FormatImage(bgImgId)
				TweenService:Create(wp.ImageLabel, TWEEN_SMOOTH, {ImageTransparency = bgImgTrans}):Play()
				TweenService:Create(wp.OverlayFrame, TWEEN_SMOOTH, {BackgroundTransparency = bgOverlayTrans}):Play()
			else
				TweenService:Create(wp.ImageLabel, TWEEN_SMOOTH, {ImageTransparency = 1}):Play()
				TweenService:Create(wp.OverlayFrame, TWEEN_SMOOTH, {BackgroundTransparency = 1}):Play()
			end
		end
	end
end

local function StartNeonLoop(target)
	if not target then return end
	local propName = pcall(function() return target.Color end) and "Color" or "BackgroundColor3"
	table.insert(KazeUI.GlowElements, {Target = target, Prop = propName})

	local function createTween()
		local tweenInfo = TweenInfo.new(1.8, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true)
		local h, s, v = Color3.toHSV(KazeUI.GlowColor)
		local targetColor = Color3.fromHSV(h, s * 0.9, math.clamp(v * 1.1, 0, 1))
		local tween = TweenService:Create(target, tweenInfo, {[propName] = targetColor})
		tween:Play()
		KazeUI.NeonTweens[target] = tween
	end
	target[propName] = KazeUI.GlowColor
	createTween()
end

local function StopNeonLoop(target)
	if not target then return end
	if KazeUI.NeonTweens[target] then
		KazeUI.NeonTweens[target]:Cancel()
		KazeUI.NeonTweens[target] = nil
	end
	for i = #KazeUI.GlowElements, 1, -1 do
		if KazeUI.GlowElements[i].Target == target then table.remove(KazeUI.GlowElements, i) end
	end
end

function KazeUI:SetGlow(color)
	KazeUI.GlowColor = color
	for _, data in ipairs(KazeUI.GlowElements) do
		local target = data.Target
		local propName = data.Prop
		if target and target.Parent then
			if KazeUI.NeonTweens[target] then KazeUI.NeonTweens[target]:Cancel() end
			target[propName] = color
			local tweenInfo = TweenInfo.new(1.8, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true)
			local h, s, v = Color3.toHSV(color)
			local targetColor = Color3.fromHSV(h, s * 0.9, 1)
			local newTween = TweenService:Create(target, tweenInfo, {[propName] = targetColor})
			newTween:Play()
			KazeUI.NeonTweens[target] = newTween
		end
	end
	for i = #KazeUI.GlowCallbacks, 1, -1 do
		local data = KazeUI.GlowCallbacks[i]
		if data.Inst and data.Inst.Parent then
			task.spawn(data.Callback, color)
		else
			table.remove(KazeUI.GlowCallbacks, i)
		end
	end
end

-- Pre-measured Responsive Height Calibration (Layout bug fixes)
local function GetTextHeights(title, body, textWidth)
	local titleHeight = 0
	if title and title ~= " " and title ~= "" then
		local ts = TextService:GetTextSize(string.upper(title), 12, Enum.Font.Code, Vector2.new(textWidth, 10000))
		titleHeight = math.ceil(ts.Y)
	end
	local bodyHeight = 0
	if body and body ~= "" then
		local bs = TextService:GetTextSize(body, 10, Enum.Font.Code, Vector2.new(textWidth, 10000))
		bodyHeight = math.ceil(bs.Y)
	end
	return titleHeight, bodyHeight
end

local function MeasureCardHeight(title, body, hasWidget)
	-- Absolute Page Width estimation based on Window base (630) - Sidebar (195) = 435px.
	-- Sizing margin is 96%, giving 417px card width.
	local maxTextWidth = 417 - CARD_PADDING_W
	if hasWidget then
		maxTextWidth = maxTextWidth - WIDGET_SPACE_W
	end
	
	local titleHeight, bodyHeight = GetTextHeights(title, body, maxTextWidth)
	local gap = (titleHeight > 0 and bodyHeight > 0) and 4 or 0
	local padding = 22 -- 11px top, 11px bottom
	return math.max(38, titleHeight + gap + bodyHeight + padding), titleHeight, bodyHeight
end

		TweenService:Create(closeBtn, TWEEN_FAST, {ImageColor3 = KazeUI.CurrentTheme.MutedText, ImageTransparency = 0.3}):Play()
	end)
	closeBtn.MouseButton1Click:Connect(dismiss)

	local autoThread = task.spawn(function()
		TweenService:Create(progressBar, TweenInfo.new(duration, Enum.EasingStyle.Linear), {Size = UDim2.new(0, 0, 1, 0)}):Play()
		task.wait(duration)
		if not isDismissing then dismiss() end
	end)
	notifHolder.Destroying:Connect(function() task.cancel(autoThread) end)
end

function KazeUI:SaveConfig(name)
	if not name or name == "" then return end
	local data = {}
	for flag, obj in pairs(KazeUI.Flags) do data[flag] = obj.Get() end
	
	if writefile then
		local s, encoded = pcall(function() return HttpService:JSONEncode(data) end)
		if s then
			local folder = KazeUI.AuthorName
			if makefolder then pcall(function() makefolder(folder) end) end
			writefile(folder .. "/" .. name .. ".json", encoded)
			PlaySound("Success")
			KazeUI:Notify({Title = "SYS PARAMS ARCHIVED", Content = "Saved structural config settings in " .. name .. ".json", Duration = 3})
		end
	end
end

function KazeUI:LoadConfig(name)
	if readfile then
		local folder = KazeUI.AuthorName
		local s, res = pcall(function() return readfile(folder .. "/" .. name .. ".json") end)
		if s then
			local s2, data = pcall(function() return HttpService:JSONDecode(res) end)
			if s2 and type(data) == "table" then
				if data["KazeUI_Theme_Preset"] ~= nil and KazeUI.Flags["KazeUI_Theme_Preset"] then
					KazeUI.Flags["KazeUI_Theme_Preset"].Set(data["KazeUI_Theme_Preset"])
				end
				for flag, val in pairs(data) do
					if flag ~= "KazeUI_Theme_Preset" and KazeUI.Flags[flag] then
						KazeUI.Flags[flag].Set(val)
					end
				end
				PlaySound("Success")
				KazeUI:Notify({Title = "PARAMS SYNCHRONIZED", Content = "Loaded interface config configurations from " .. name .. ".json", Duration = 3})
			end
		end
	end
end

local function ToggleScrollingAncestors(element, state)
	local current = element.Parent
	while current do
		if current:IsA("ScrollingFrame") then current.ScrollingEnabled = state end
		current = current.Parent
	end
end

			end

			elseif count == 1 then initialText = string.upper(first)
			else initialText = count .. " SELECT" end

			initialText = string.upper(tostring(selectedSingle))
		end
	end

	selectedLabel.Size = UDim2.new(0.5, -24, 1, 0)

	selectedLabel.BackgroundTransparency = 1
	selectedLabel.Font = Enum.Font.Code
	selectedLabel.TextSize = 11
	selectedLabel.TextXAlignment = Enum.TextXAlignment.Right
	selectedLabel.TextTruncate = Enum.TextTruncate.AtEnd

	local clickBtn = Instance.new("TextButton", header)
	clickBtn.Size = UDim2.fromScale(1, 1)
	clickBtn.BackgroundTransparency = 1
	clickBtn.Text = ""

	local searchWrapper = Instance.new("Frame")
	searchWrapper.Size = UDim2.new(1, -20, 0, 28)
	searchWrapper.BackgroundTransparency = 1
	searchWrapper.LayoutOrder = 2
	searchWrapper.Visible = false
	searchWrapper.Parent = dropFrame

	searchBox.BorderSizePixel = 0

	searchBox.Text = ""
	searchBox.Font = Enum.Font.Code
	searchBox.TextSize = 11
	searchBox.TextXAlignment = Enum.TextXAlignment.Left
	searchBox.Parent = searchWrapper
	Instance.new("UICorner", searchBox).CornerRadius = UDim.new(0, 3)
	KazeUI:AddPanel(searchBox, -0.02)
	KazeUI:RegisterText(searchBox, false)

	local searchStroke = Instance.new("UIStroke", searchBox)
	searchStroke.Thickness = 1
	KazeUI:RegisterBorder(searchStroke)

	local searchPadding = Instance.new("UIPadding", searchBox)
	searchPadding.PaddingLeft = UDim.new(0, 8)

	local divider = Instance.new("Frame")
	divider.Size = UDim2.new(1, -20, 0, 1)
	divider.BorderSizePixel = 0
	divider.LayoutOrder = 3
	divider.Visible = false
	divider.Parent = dropFrame
	KazeUI:OnThemeChanged(divider, function(t) divider.BackgroundColor3 = t.Border end)

	local listContainer = Instance.new("ScrollingFrame")
	listContainer.Size = UDim2.new(1, -20, 0, 0)
	listContainer.BackgroundTransparency = 1
	listContainer.BorderSizePixel = 0
	listContainer.LayoutOrder = 4
	listContainer.Visible = false
	listContainer.ScrollBarThickness = 2
	listContainer.ScrollBarImageColor3 = Color3.fromRGB(40, 80, 110)
	listContainer.Parent = dropFrame

	listContainer.MouseEnter:Connect(function() ToggleScrollingAncestors(dropFrame, false) end)
	listContainer.MouseLeave:Connect(function() ToggleScrollingAncestors(dropFrame, true) end)

	local listLayout = Instance.new("UIListLayout", listContainer)
	listLayout.SortOrder = Enum.SortOrder.LayoutOrder
	listLayout.Padding = UDim.new(0, 3)

	local optionItems = {}
	local function CalcHeight()
		local count = 0
		for _, item in ipairs(optionItems) do if item.Btn.Visible then count = count + 1 end end
		local targetH = math.min(count, 5) * 32
		TweenService:Create(listContainer, TWEEN_FAST, {Size = UDim2.new(1, -20, 0, targetH)}):Play()
		listContainer.CanvasSize = UDim2.new(0, 0, 0, count * 32)
	end

	local function CloseDropdown()
		isOpen = false
		searchBox.Text = ""
		searchWrapper.Visible = false
		divider.Visible = false
		listContainer.Visible = false
		TweenService:Create(arrowIcon, TWEEN_SPRING, {Rotation = 0}):Play()
		TweenService:Create(boxStroke, TWEEN_FAST, {Color = KazeUI.CurrentTheme.Border}):Play()
	end

	local function Refresh(newOptions)
		for _, item in ipairs(optionItems) do item.Btn:Destroy() end
		table.clear(optionItems)

		options = type(newOptions) == "table" and newOptions or {}
		for _, opt in ipairs(options) do
			local optStr = tostring(opt)
			local btn = Instance.new("TextButton")
			btn.Size = UDim2.new(1, -4, 0, 28)
			btn.BorderSizePixel = 0
			btn.Text = "  " .. string.upper(optStr)
			btn.Font = Enum.Font.Code
			btn.TextSize = 11
			btn.TextXAlignment = Enum.TextXAlignment.Left
			btn.Parent = listContainer
			Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 3)
			KazeUI:AddPanel(btn, 0.04, true)

			local indicator
			if isMulti then
				indicator = Instance.new("Frame", btn)
				indicator.AnchorPoint = Vector2.new(1, 0.5)
				indicator.Position = UDim2.new(1, -10, 0.5, 0)
				indicator.Size = UDim2.fromOffset(10, 10)
				local indStroke = Instance.new("UIStroke", indicator)
				KazeUI:RegisterBorder(indStroke)
				Instance.new("UICorner", indicator).CornerRadius = UDim.new(0, 2)
				KazeUI:AddPanel(indicator, 0.06)
			end

			local isHovering = false
			local function UpdateVisuals()
				local active = isMulti and selectedMulti[optStr] or (not isMulti and selectedSingle == optStr)
				local targetBg = active and KazeUI.GetColor(0.08) or KazeUI.GetColor(0.04)
				local targetTxt = active and KazeUI.CurrentTheme.Text or KazeUI.CurrentTheme.MutedText
				local indBg = active and KazeUI.GlowColor or KazeUI.GetColor(0.04)

				if isHovering then
					targetBg = KazeUI.GetColor(0.12)
					targetTxt = KazeUI.CurrentTheme.Text
				end

				btn.BackgroundColor3 = targetBg
				btn.TextColor3 = targetTxt
				if isMulti and indicator then indicator.BackgroundColor3 = indBg end
			end
			UpdateVisuals()

			btn.MouseEnter:Connect(function()
				isHovering = true
				PlaySound("Hover")
				UpdateVisuals()
			end)
			btn.MouseLeave:Connect(function()
				isHovering = false
				UpdateVisuals()
			end)

			btn.MouseButton1Click:Connect(function()
				PlaySound("Click")
				if isMulti then
					selectedMulti[optStr] = not selectedMulti[optStr]
					UpdateVisuals()
					rebuildInitText()
					selectedLabel.Text = initialText
					PushConsoleLog(GetTimestamp() .. " [INPUT] Multidropdown sync: " .. optStr)
					if type(callback) == "function" then
						local r = {}
						for k, v in pairs(selectedMulti) do if v then table.insert(r, k) end end
						task.spawn(callback, r)
					end
				else
					selectedSingle = optStr
					rebuildInitText()
					selectedLabel.Text = initialText
					CloseDropdown()
					for _, item in ipairs(optionItems) do item.Update() end
					PushConsoleLog(GetTimestamp() .. " [INPUT] Dropdown: " .. optStr)
					if type(callback) == "function" then task.spawn(callback, selectedSingle) end
				end
			end)

			table.insert(optionItems, {Btn = btn, Text = optStr, Update = UpdateVisuals})
		end
		CalcHeight()
	end
	Refresh(options)

	searchBox:GetPropertyChangedSignal("Text"):Connect(function()
		local q = string.lower(searchBox.Text)
		for _, item in ipairs(optionItems) do
			item.Btn.Visible = (q == "" or string.find(string.lower(item.Text), q, 1, true)) ~= nil
		end
		CalcHeight()
	end)

	clickBtn.MouseButton1Click:Connect(function()
		PlaySound("Click")
		isOpen = not isOpen
		if isOpen then
			searchWrapper.Visible = true
			divider.Visible = true
			listContainer.Visible = true
			CalcHeight()
			TweenService:Create(arrowIcon, TWEEN_SPRING, {Rotation = 180}):Play()
			TweenService:Create(boxStroke, TWEEN_FAST, {Color = KazeUI.GlowColor}):Play()
		else
			CloseDropdown()
		end
	end)

	RegisterFlag(flag, function()
		if isMulti then
			local r = {}
			for k, v in pairs(selectedMulti) do if v then table.insert(r, k) end end
			return r
		end
		return selectedSingle
	end, function(val)
		if isMulti then
			selectedMulti = {}
			if type(val) == "table" then
				for _, v in ipairs(val) do selectedMulti[tostring(v)] = true end
			end
		else
			selectedSingle = tostring(val)
		end
		rebuildInitText()
		selectedLabel.Text = initialText
		Refresh(options)
	end)

	return {
		Wrapper = wrapper,
		Frame = dropFrame,
		Refresh = function(self, newOpts) Refresh(newOpts) end,
		GetValue = function()
			if isMulti then
				local r = {}
				for k, v in pairs(selectedMulti) do if v then table.insert(r, k) end end
				return r
			end
			return selectedSingle
		end,
		SetText = function(self, newTitle) titleLabel.Text = string.upper(newTitle) end
	}
end

		KazeUI:RegisterText(ptitleLabel, false)

	valueLabel.Size = UDim2.new(0, 70, 0, titleHeight > 0 and titleHeight or 16)
	valueLabel.Position = UDim2.new(1, -12, 0, 11)
	valueLabel.AnchorPoint = Vector2.new(1, 0)

	valueLabel.Text = string.format("%.1f", value)
	valueLabel.Font = Enum.Font.Code
	valueLabel.TextSize = 11
	valueLabel.TextXAlignment = Enum.TextXAlignment.Right
	KazeUI:RegisterText(valueLabel, false)

	track.Size = UDim2.new(1, -24, 0, 6)
	track.Position = UDim2.fromOffset(12, 11 + titleHeight + 8)
	track.BorderSizePixel = 0
	Instance.new("UICorner", track).CornerRadius = UDim.new(0, 1)
	KazeUI:AddPanel(track, -0.01)
	
	local fill = Instance.new("Frame", track)
	fill.Size = UDim2.new(0, 0, 1, 0)
	fill.BackgroundColor3 = KazeUI.GlowColor
	fill.BorderSizePixel = 0
	
	local knob = Instance.new("Frame", track)
	knob.AnchorPoint = Vector2.new(0.5, 0.5)
	knob.Size = UDim2.fromOffset(8, 14) 
	knob.Position = UDim2.new(0, 0, 0.5, 0)
	knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	knob.BorderSizePixel = 0
	Instance.new("UICorner", knob).CornerRadius = UDim.new(0, 2)
	
	local knobStroke = Instance.new("UIStroke", knob)
	knobStroke.Thickness = 1
	knobStroke.Color = KazeUI.GlowColor

	local clickDetector = Instance.new("TextButton", track)
	clickDetector.Size = UDim2.new(1, 0, 1, 14)
	clickDetector.Position = UDim2.new(0, 0, 0, -4)
	clickDetector.BackgroundTransparency = 1
	clickDetector.Text = ""
	
	local isDragging = false
	local function UpdateValue(input)
		local trackWidth = track.AbsoluteSize.X
		if trackWidth <= 0 then return end
		local relativeX = input.Position.X - track.AbsolutePosition.X
		local percentage = math.clamp(relativeX / trackWidth, 0, 1)
		local newVal = min + (max - min) * percentage
		newVal = math.clamp(newVal, min, max)
		value = newVal
		valueLabel.Text = string.format("%.1f", value)
		
		local visualPercentage = (value - min) / (max - min)
		fill.Size = UDim2.new(visualPercentage, 0, 1, 0)
		knob.Position = UDim2.new(visualPercentage, 0, 0.5, 0)
		if typeof(callback) == "function" then task.spawn(callback, value) end
	end
	
	local startPercent = (value - min) / (max - min)
	fill.Size = UDim2.new(startPercent, 0, 1, 0)
	knob.Position = UDim2.new(startPercent, 0, 0.5, 0)
	
	clickDetector.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			isDragging = true
			PlaySound("Hover")
			ToggleScrollingAncestors(track, false)
			LockCamera()
			UpdateValue(input)
			TweenService:Create(paraStroke, TWEEN_FAST, {Color = KazeUI.GlowColor}):Play()
			
			local move, ends
			move = UIS.InputChanged:Connect(function(mv)
				if isDragging and (mv.UserInputType == Enum.UserInputType.MouseMovement or mv.UserInputType == Enum.UserInputType.Touch) then
					UpdateValue(mv)
				end
			end)
			ends = UIS.InputEnded:Connect(function(ed)
				if ed.UserInputType == Enum.UserInputType.MouseButton1 or ed.UserInputType == Enum.UserInputType.Touch then
					isDragging = false
					PlaySound("Click")
					ToggleScrollingAncestors(track, true)
					UnlockCamera()
					TweenService:Create(paraStroke, TWEEN_FAST, {Color = KazeUI.CurrentTheme.Border}):Play()
					PushConsoleLog(GetTimestamp() .. " [PARAM] Slider: " .. string.format("%.1f", value))
					if move then move:Disconnect() end
					if ends then ends:Disconnect() end
				end
			end)
		end
	end)

	local api = {
		Wrapper = wrapper,
		SetValue = function(self, newVal)
			value = math.clamp(tonumber(newVal) or min, min, max)
			valueLabel.Text = string.format("%.1f", value)
			local valPercent = (value - min) / (max - min)
			fill.Size = UDim2.new(valPercent, 0, 1, 0)
			knob.Position = UDim2.new(valPercent, 0, 0.5, 0)
			if typeof(callback) == "function" then task.spawn(callback, value) end
		end,
		SetText = function(self, newTitle)
			if ptitleLabel and newTitle then ptitleLabel.Text = string.upper(newTitle) end
		end
	}
	RegisterFlag(flag, function() return value end, function(val) api:SetValue(val) end)
	return api
end

	local bindContainer = Instance.new("TextButton", paraFrame)
	bindContainer.Size = UDim2.fromOffset(76, 22)
	bindContainer.AnchorPoint = Vector2.new(1, 0.5)
	bindContainer.Position = UDim2.new(1, -12, 0.5, 0)
	bindContainer.BorderSizePixel = 0
	bindContainer.Text = ""
	bindContainer.AutoButtonColor = false
	Instance.new("UICorner", bindContainer).CornerRadius = UDim.new(0, 3)
	KazeUI:AddPanel(bindContainer, 0.05)

	local bindStroke = Instance.new("UIStroke", bindContainer)
	bindStroke.Thickness = 1
	KazeUI:RegisterBorder(bindStroke)
	
	local bindLabel = Instance.new("TextLabel", bindContainer)
	bindLabel.Size = UDim2.fromScale(1, 1)
	bindLabel.BackgroundTransparency = 1
	bindLabel.Text = currentKey == Enum.KeyCode.Unknown and "NONE" or string.upper(currentKey.Name)
	bindLabel.Font = Enum.Font.Code
	bindLabel.TextSize = 10
	KazeUI:RegisterText(bindLabel, false)

	local isHovering = false
	bindContainer.MouseEnter:Connect(function()
		isHovering = true
		PlaySound("Hover")
		if not isBinding then TweenService:Create(bindStroke, TWEEN_FAST, {Color = KazeUI.GlowColor}):Play() end
	end)
	bindContainer.MouseLeave:Connect(function()
		isHovering = false
		if not isBinding then TweenService:Create(bindStroke, TWEEN_FAST, {Color = KazeUI.CurrentTheme.Border}):Play() end
	end)
	
	bindContainer.MouseButton1Click:Connect(function()
		if isBinding then return end

		bindLabel.Text = "WAITING..."
		TweenService:Create(bindStroke, TWEEN_FAST, {Color = KazeUI.GlowColor}):Play()
		PushConsoleLog(GetTimestamp() .. " [INPUT] Binding sensor key...")
		
		local conn
		conn = UIS.InputBegan:Connect(function(input)
			local isKey = input.UserInputType == Enum.UserInputType.Keyboard
			local isMouse = input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.MouseButton2 or input.UserInputType == Enum.UserInputType.MouseButton3
			
			if isKey or isMouse then
				if isKey then
					local key = input.KeyCode
					if key == Enum.KeyCode.Escape then
					elseif key == Enum.KeyCode.Backspace or key == Enum.KeyCode.Delete then
						currentKey = Enum.KeyCode.Unknown
					else
						currentKey = key
					end
				end
				PlaySound("Success")
				bindLabel.Text = currentKey == Enum.KeyCode.Unknown and "NONE" or string.upper(currentKey.Name)
				TweenService:Create(bindStroke, TWEEN_FAST, {Color = KazeUI.CurrentTheme.Border}):Play()
				PushConsoleLog(GetTimestamp() .. " [INPUT] Sensor bound to: " .. currentKey.Name)
				conn:Disconnect()
				task.delay(0.15, function() isBinding = false end)
			end
		end)
	end)
	
	UIS.InputBegan:Connect(function(input, gp)
		if not gp and not isBinding and currentKey ~= Enum.KeyCode.Unknown and input.KeyCode == currentKey then
			PushConsoleLog(GetTimestamp() .. " [EXEC] Sensor key active: " .. currentKey.Name)
			if typeof(callback) == "function" then task.spawn(callback, currentKey) end
		end
	end)

	RegisterFlag(flag, function() return currentKey.Name end, function(val)
		pcall(function()
			currentKey = Enum.KeyCode[val] or Enum.KeyCode.Unknown
			bindLabel.Text = currentKey == Enum.KeyCode.Unknown and "NONE" or string.upper(currentKey.Name)
		end)
	end)

	return {
		Wrapper = wrapper,
		SetText = function(self, newTitle)
			if ptitleLabel and newTitle then ptitleLabel.Text = string.upper(newTitle) end
		end,
		SetKey = function(self, newKey)
			currentKey = newKey
			bindLabel.Text = currentKey == Enum.KeyCode.Unknown and "NONE" or string.upper(currentKey.Name)
		end
	}
end

AttachElementsToAPI = function(apiTable, parentFrame)
	function apiTable:AddLabel(arg1)
		local t = type(arg1) == "table" and (arg1.Text or arg1.Title) or arg1
		local label = Instance.new("TextLabel")
		label.Size = UDim2.new(1, -12, 0, 22)
		label.BackgroundTransparency = 1
		label.Text = "> " .. string.upper(tostring(t or " "))
		label.Font = Enum.Font.Code
		label.TextSize = 11
		label.TextXAlignment = Enum.TextXAlignment.Left
		label.TextWrapped = true
		label.Parent = parentFrame
		KazeUI:RegisterText(label, false)
		return label
	end

	function apiTable:Paragraph(arg1, arg2)
		local title = type(arg1) == "table" and arg1.Title or arg1
		local content = type(arg1) == "table" and (arg1.Content or arg1.Description) or arg2
		return CreateParagraph(parentFrame, title, content)
	end

	function apiTable:Button(arg1, arg2, arg3)
		local title = type(arg1) == "table" and arg1.Title or arg1
		local desc = type(arg1) == "table" and (arg1.Description or arg1.Content or "") or arg2
		local callback = type(arg1) == "table" and arg1.Callback or arg3
		return CreateClickableParagraph(parentFrame, title, desc, callback)
	end

	function apiTable:Toggle(arg1, arg2, arg3, arg4, arg5)
		local title, desc, def, callback, flag = "", "", false, nil, nil
		if type(arg1) == "table" then
			title, desc, def, callback, flag = arg1.Title, (arg1.Description or arg1.Content or ""), arg1.Default, arg1.Callback, arg1.Flag
		else
			title, desc, def, callback, flag = arg1, arg2, arg3, arg4, arg5
		end
		return CreateToggleSwitch(parentFrame, title, desc, def, callback, flag)
	end

	function apiTable:TextBox(arg1, arg2, arg3, arg4, arg5)
		local title, place, def, callback, flag = "", "", "", nil, nil
		if type(arg1) == "table" then
			title, place, def, callback, flag = arg1.Title, arg1.Placeholder, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, place, def, callback, flag = arg1, arg2, arg3, arg4, arg5
		end
		return CreateTextBoxInput(parentFrame, title, place, def, callback, flag)
	end

	function apiTable:Slider(arg1, arg2, arg3, arg4, arg5, arg6)
		local title, min, max, def, callback, flag = "", 0, 100, 50, nil, nil
		if type(arg1) == "table" then
			title, min, max, def, callback, flag = arg1.Title, arg1.Min, arg1.Max, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, min, max, def, callback, flag = arg1, arg2, arg3, arg4, arg5, arg6
		end
		return CreateSliderInput(parentFrame, title, min, max, def, callback, flag)
	end

	function apiTable:Dropdown(arg1, arg2, arg3, arg4, arg5, arg6)
		local title, vals, def, callback, multi, flag = "", {}, nil, nil, false, nil
		if type(arg1) == "table" then
			title, vals, def, multi, callback, flag = arg1.Title, (arg1.Values or arg1.Options), arg1.Default, (arg1.Multi or false), arg1.Callback, arg1.Flag
		else
			title, vals, def, callback, multi, flag = arg1, arg2, arg3, arg4, (arg5 or false), arg6
		end
		return CreateDropdown(parentFrame, title, vals, def, callback, multi, flag)
	end

	function apiTable:ColorPicker(arg1, arg2, arg3, arg4)
		local title, def, cb, flag = "", Color3.fromRGB(255, 255, 255), nil, nil
		if type(arg1) == "table" then
			title, def, cb, flag = arg1.Title, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, def, cb, flag = arg1, arg2, arg3, arg4
		end
		return CreateColorPicker(parentFrame, title, def, cb, flag)
	end
	
	function apiTable:Keybind(arg1, arg2, arg3, arg4)
		local title, def, cb, flag = "", Enum.KeyCode.Unknown, nil, nil
		if type(arg1) == "table" then
			title, def, cb, flag = arg1.Title, arg1.Default, arg1.Callback, arg1.Flag
		else
			title, def, cb, flag = arg1, arg2, arg3, arg4
		end
		return CreateKeybindInput(parentFrame, title, def, cb, flag)
	end

	function apiTable:SectionUI(arg1)
		local title = type(arg1) == "table" and arg1.Title or arg1
		return CreateCollapsibleSection(parentFrame, title)
	end
	
	function apiTable:Section(arg1)
		local title = type(arg1) == "table" and arg1.Title or arg1
		return CreateVisualSectionDivider(parentFrame, title)
	end
end

	contentPadding.PaddingBottom = UDim.new(0, 8)
	contentPadding.PaddingLeft = UDim.new(0, 4)
	contentPadding.PaddingRight = UDim.new(0, 4)

	toggleDetector.Size = UDim2.fromScale(1, 1)
	toggleDetector.BackgroundTransparency = 1
	toggleDetector.Text = ""

		isOpen = not isOpen
		sectionContent.Visible = isOpen
		divider.Visible = isOpen

		local animDur = 0.2
		if isOpen then
			TweenService:Create(arrowIcon, TweenInfo.new(animDur), {Rotation = 0, ImageColor3 = KazeUI.GlowColor}):Play()
			TweenService:Create(accentLine, TweenInfo.new(animDur), {BackgroundColor3 = KazeUI.GlowColor}):Play()

			StartNeonLoop(accentLine)
			PushConsoleLog(GetTimestamp() .. " [SYS] Module expanded: " .. title)
		else
			TweenService:Create(arrowIcon, TweenInfo.new(animDur), {Rotation = -90, ImageColor3 = KazeUI.CurrentTheme.MutedText}):Play()
			TweenService:Create(accentLine, TweenInfo.new(animDur), {BackgroundColor3 = KazeUI.CurrentTheme.MutedText}):Play()
			StopNeonLoop(boxStroke)
			StopNeonLoop(accentLine)
			boxStroke.Color = KazeUI.CurrentTheme.Border
			accentLine.BackgroundColor3 = KazeUI.CurrentTheme.MutedText
		end
	end

		PlaySound("Hover")
		TweenService:Create(headerFrame, TweenInfo.new(0.12), {BackgroundColor3 = KazeUI.GetColor(0.04), BackgroundTransparency = 0.5}):Play()

	AttachElementsToAPI(sectionPublic, sectionContent)

	function Win:CreateConfigManager(tab)
		local sec = tab:SectionUI("Stored Hardware Matrices")
		local cfgName = "Aegis_Params"

		local function getConfigs()
			local list = {}
			local folder = KazeUI.AuthorName
			if makefolder then pcall(function() makefolder(folder) end) end
			if listfiles then
				local s, files = pcall(function() return listfiles(folder) end)
				if s and type(files) == "table" then
					for _, f in ipairs(files) do
						local fileName = string.match(f, "([^/\\]+)$") or f
						local name = string.match(fileName, "^(.+)%.json$")
						if name then table.insert(list, name) end
					end
				end
			end
			if #list == 0 then table.insert(list, "Aegis_Params") end
			return list
		end

		sec:TextBox({Title = "Matrix State Sector", Default = "Aegis_Params", Callback = function(v) cfgName = v end})
		
		local configDropdown
		sec:Button({Title = "Archive Spatial Coordinates", Callback = function() 
			KazeUI:SaveConfig(cfgName) 
			if configDropdown then configDropdown:Refresh(getConfigs()) end
		end})

		configDropdown = sec:Dropdown({
			Title = "System Presets", 
			Options = getConfigs(), 
			Callback = function(v) cfgName = v end
		})

		sec:Button({Title = "Scan Local Matrices", Callback = function() if configDropdown then configDropdown:Refresh(getConfigs()) end end})
		sec:Button({Title = "Inject Selected Segment", Callback = function() KazeUI:LoadConfig(cfgName) end})
		
		sec:Button({Title = "Purge Segment File", Callback = function()
			if delfile then
				local folder = KazeUI.AuthorName
				pcall(function() delfile(folder .. "/" .. cfgName .. ".json") end)
				KazeUI:Notify({Title = "DATA CORRUPTED", Content = "Purged segment bank: " .. cfgName, Duration = 3})
				if configDropdown then configDropdown:Refresh(getConfigs()) end
			end
		end})
	end

			if not isStateVisible then return end
			isStateVisible = false

			tw:Play()
			tw.Completed:Connect(function() 
				if not isStateVisible then EmptyState.Visible = false end

		end

		local function updateState()
			local count = 0
			for _, child in ipairs(pageContent:GetChildren()) do
				if child:IsA("GuiObject") and child.Visible then count = count + 1 end
			end
			if count == 0 then showEmpty() else hideEmpty() end
		end

		pageContent.ChildAdded:Connect(function(child)
			if child:IsA("GuiObject") then
				child:GetPropertyChangedSignal("Visible"):Connect(updateState)
				updateState()
			end
		end)

			end

			isHovering = false
			if not leftIndicator.Visible then

				TweenService:Create(txt, TWEEN_FAST, {TextColor3 = KazeUI.CurrentTheme.MutedText}):Play()
				TweenService:Create(iconLabel, TWEEN_FAST, {ImageColor3 = KazeUI.CurrentTheme.MutedText}):Play()
			end
		end)

		if #self._Tabs == 1 then activate() end

		KazeUI:OnThemeChanged(btn, function(t)
			if leftIndicator.Visible then
				txt.TextColor3 = t.Text
				if not isHovering then iconLabel.ImageColor3 = KazeUI.GlowColor end
				btn.BackgroundColor3 = KazeUI.GetColor(0.04)
			else
				txt.TextColor3 = t.MutedText
				if not isHovering then iconLabel.ImageColor3 = t.MutedText end
				btn.BackgroundColor3 = KazeUI.GetColor(0.02)
			end
		end)

		AttachElementsToAPI(public, pageContent)

		return public
	end

	BootLabel.Font = Enum.Font.Code
	BootLabel.TextSize = 12
	BootLabel.TextColor3 = Color3.fromRGB(0, 210, 255)

		local phases = {

		for _, text in ipairs(phases) do
			BootLabel.Text = "AEGIS COGNITIVE INTERFACE [v4.5]\n\n" .. text
			task.wait(0.3)
		end
		TweenService:Create(BootShield, TweenInfo.new(0.4), {BackgroundTransparency = 1}):Play()
		TweenService:Create(BootLabel, TweenInfo.new(0.4), {TextTransparency = 1}):Play()

	end)

	table.insert(KazeUI.ActiveWindows, Window)
	if typeof(config.Callback) == "function" then pcall(config.Callback) end
