local function UpdateScale()
	local size = workspace.CurrentCamera.ViewportSize
	local width, height = size.X, size.Y
	local minAxis = math.min(width, height)
	local targetScale = 1.0

	if UIS.TouchEnabled then
		targetScale = (minAxis < 500) and 0.85 or 1.0
	else
		if width >= 1920 or height >= 1080 then
			targetScale = 1.05
		elseif width <= 1366 then
			targetScale = 0.95
		else
			targetScale = 1.0
		end
	end

	UIScale.Scale = targetScale
	ScreenGui.IgnoreGuiInset = true
end

UpdateScale()
workspace.CurrentCamera:GetPropertyChangedSignal("ViewportSize"):Connect(UpdateScale)

-- Unified Lucide Icon Parser
local IconsMap = {}
pcall(function()
	if game and game.HttpGet then
		local rawIcons = game:HttpGet("https://raw.githubusercontent.com/KazeHub/UI/refs/heads/main/Icons.lua.txt")
		if rawIcons then
			local s, res = pcall(function()
				local func = loadstring(rawIcons)
				return func and func()
			end)
			if not s or type(res) ~= "table" then
				s, res = pcall(function() return HttpService:JSONDecode(rawIcons) end)
			end
			if type(res) == "table" then
				for k, v in pairs(res) do IconsMap[string.lower(tostring(k))] = tostring(v) end
			end
		end
	end
end)

local function FormatImage(id)
	if not id or id == "" then return "" end
	id = tostring(id)
	local lowerId = string.lower(id)
	if IconsMap[lowerId] then
		local mapped = IconsMap[lowerId]
		if not string.find(mapped, "rbxassetid://") and not string.find(mapped, "http") then
			local clean = string.gsub(mapped, "%D", "")
			if clean ~= "" then return "rbxassetid://" .. clean end
		end
		return mapped
	end
	if string.find(id, "rbxassetid://") or string.find(id, "http") then return id end
	local clean = string.gsub(id, "%D", "")
	if clean == "" then return id end
	return "rbxassetid://" .. clean
end

-- Theme & Registration Bindings
function KazeUI:RegisterText(inst, isMuted)
	if not inst then return end
	if isMuted then
		table.insert(KazeUI.MutedTextElements, inst)
		inst.TextColor3 = KazeUI.CurrentTheme.MutedText
		if inst:IsA("TextBox") then inst.PlaceholderColor3 = KazeUI.CurrentTheme.MutedText end
	else
		table.insert(KazeUI.TextElements, inst)
		inst.TextColor3 = KazeUI.CurrentTheme.Text
	end
	inst.Destroying:Connect(function()
		local tbl = isMuted and KazeUI.MutedTextElements or KazeUI.TextElements
		for i = #tbl, 1, -1 do
			if tbl[i] == inst then table.remove(tbl, i) end
		end
	end)
end

function KazeUI:RegisterBorder(inst)
	if not inst then return end
	table.insert(KazeUI.BorderElements, inst)
	inst.Color = KazeUI.CurrentTheme.Border
	inst.Destroying:Connect(function()
		for i = #KazeUI.BorderElements, 1, -1 do
			if KazeUI.BorderElements[i] == inst then table.remove(KazeUI.BorderElements, i) end
		end
	end)
end

function KazeUI:OnThemeChanged(inst, cb)
	if not inst then return end
	table.insert(KazeUI.ThemeCallbacks, {Inst = inst, Callback = cb})
	task.spawn(cb, KazeUI.CurrentTheme)
	inst.Destroying:Connect(function()
		for i = #KazeUI.ThemeCallbacks, 1, -1 do
			if KazeUI.ThemeCallbacks[i].Inst == inst then table.remove(KazeUI.ThemeCallbacks, i) end
		end
	end)
end

-- Dragging Controller
local lastNormalPosition = UDim2.fromScale(0.5, 0.5)
local lastNormalAnchor = Vector2.new(0.5, 0.5)

local function MakeDraggable(dragPart, targetPart)
	local dragging = false
	local dragInput, dragStart, startPos
	local dragActive = false
	
	dragPart.Active = true
	
	local function update(input)
		if not dragStart or not startPos then return end
		local delta = input.Position - dragStart
		local scale = UIScale.Scale
		
		if dragActive or delta.Magnitude > 8 then
			dragActive = true
			targetPart.Position = UDim2.new(
				startPos.X.Scale, startPos.X.Offset + (delta.X / scale),
				startPos.Y.Scale, startPos.Y.Offset + (delta.Y / scale)
			)
			if targetPart:IsA("Frame") then
				lastNormalPosition = targetPart.Position
				lastNormalAnchor = targetPart.AnchorPoint
			end
		end
	end
	
	dragPart.InputBegan:Connect(function(input)
		if KazeUI.IsResizing then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragActive = false
			dragStart = input.Position
			startPos = targetPart.Position
			LockCamera()
			
			local changeCon
			changeCon = input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
					dragInput = nil
					UnlockCamera()
					if changeCon then changeCon:Disconnect() end
				end
			end)
		end
	end)
	
	dragPart.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	
	local uisConn = UIS.InputChanged:Connect(function(input)
		if input == dragInput and dragging then update(input) end
	end)
	
	dragPart.Destroying:Connect(function()
		if uisConn then uisConn:Disconnect() end
	end)
end

-- Clean Background and Depth System
function KazeUI:AddPanel(inst, elevation, ignoreColor, ignoreTransparency)
	table.insert(KazeUI.PanelElements, {
		Inst = inst,
		Elevation = elevation or 0,
		IgnoreColor = ignoreColor,
		IgnoreTransparency = ignoreTransparency
	})
	if inst and inst:IsA("GuiObject") then
		if not ignoreColor then
			inst.BackgroundColor3 = (elevation and elevation > 0) and KazeUI.CurrentTheme.Panel or KazeUI.CurrentTheme.Background
		end
		if not ignoreTransparency then
			inst.BackgroundTransparency = KazeUI.BackgroundTransparency
		end
		inst.Destroying:Connect(function()
			for i = #KazeUI.PanelElements, 1, -1 do
				if KazeUI.PanelElements[i].Inst == inst then table.remove(KazeUI.PanelElements, i) end
			end
		end)
	end
end

function KazeUI:SetTransparency(alpha)
	KazeUI.BackgroundTransparency = alpha
	for _, data in ipairs(KazeUI.PanelElements) do
		if data.Inst and data.Inst.Parent and not data.IgnoreTransparency then
			data.Inst.BackgroundTransparency = alpha
		end
	end
end

function KazeUI:SetTheme(themeName)
	local targetTheme = KazeUI.Themes[themeName] or KazeUI.Themes.Obsidian
	KazeUI.CurrentTheme = targetTheme
	KazeUI.BackgroundColor = targetTheme.Background
	
	for _, data in ipairs(KazeUI.PanelElements) do
		if data.Inst and data.Inst.Parent and not data.IgnoreColor then
			data.Inst.BackgroundColor3 = (data.Elevation > 0) and targetTheme.Panel or targetTheme.Background
		end
	end
	
	for _, inst in ipairs(KazeUI.TextElements) do
		if inst and inst.Parent then inst.TextColor3 = targetTheme.Text end
	end
	for _, inst in ipairs(KazeUI.MutedTextElements) do
		if inst and inst.Parent then
			inst.TextColor3 = targetTheme.MutedText
			if inst:IsA("TextBox") then inst.PlaceholderColor3 = targetTheme.MutedText end
		end
	end
	for _, inst in ipairs(KazeUI.BorderElements) do
		if inst and inst.Parent then inst.Color = targetTheme.Border end
	end
	for _, data in ipairs(KazeUI.ThemeCallbacks) do
		if data.Inst and data.Inst.Parent then pcall(data.Callback, targetTheme) end
	end
end

function KazeUI:SaveConfig(name)
	if not name or name == "" then return end
	local data = {}
	for flag, obj in pairs(KazeUI.Flags) do data[flag] = obj.Get() end
	if writefile then
		local s, encoded = pcall(function() return HttpService:JSONEncode(data) end)
		if s then
			local folder = KazeUI.AuthorName or "KazeUI"
			if makefolder then pcall(function() makefolder(folder) end) end
			writefile(folder .. "/" .. name .. ".json", encoded)
			KazeUI:Notify({Title = "Config Saved", Content = name .. " successfully created."})
		end
	end
end

function KazeUI:LoadConfig(name)
	if readfile then
		local folder = KazeUI.AuthorName or "KazeUI"
		local s, res = pcall(function() return readfile(folder .. "/" .. name .. ".json") end)
		if s then
			local s2, data = pcall(function() return HttpService:JSONDecode(res) end)
			if s2 and type(data) == "table" then
				if data["KazeUI_Theme_Preset"] and KazeUI.Flags["KazeUI_Theme_Preset"] then
					KazeUI.Flags["KazeUI_Theme_Preset"].Set(data["KazeUI_Theme_Preset"])
				end
				for flag, val in pairs(data) do
					if flag ~= "KazeUI_Theme_Preset" and KazeUI.Flags[flag] then
						KazeUI.Flags[flag].Set(val)
					end
				end
				KazeUI:Notify({Title = "Config Loaded", Content = "Restored configurations for " .. name})
			end
		end
	end
end

local function ToggleScrollingAncestors(element, state)
	local current = element.Parent
	while current do
		if current:IsA("ScrollingFrame") then current.ScrollingEnabled = state end
		current = current.Parent
	end
end

		TweenService:Create(listContainer, TWEEN_FAST, {Size = UDim2.new(1, -16, 0, targetH)}):Play()
		listContainer.CanvasSize = UDim2.new(0, 0, 0, visibleCount * 28)
	end

	local function UpdateHeader()
		if isMulti then
			local count = 0
			local first = ""
			for k, v in pairs(selectedMulti) do if v then count = count + 1 if first == "" then first = k end end end
			if count == 0 then selectedLabel.Text = "None"
			elseif count == 1 then selectedLabel.Text = first
			else selectedLabel.Text = count .. " selected" end
		else
			selectedLabel.Text = tostring(selectedSingle)
		end
	end
	UpdateHeader()

	local function Close()
		isOpen = false
		searchBox.Text = ""
		searchWrapper.Visible = false
		listContainer.Visible = false
		TweenService:Create(arrowIcon, TWEEN_FAST, {Rotation = 0}):Play()
		TweenService:Create(boxStroke, TWEEN_FAST, {Color = KazeUI.CurrentTheme.Border}):Play()
	end

	local function Refresh(newOpts)
		for _, item in ipairs(optionItems) do item.Btn:Destroy() end
		table.clear(optionItems)
		options = type(newOpts) == "table" and newOpts or {}

		for _, opt in ipairs(options) do
			local optStr = tostring(opt)
			local btn = Instance.new("TextButton", listContainer)
			btn.Size = UDim2.new(1, 0, 0, 26)
			btn.BackgroundTransparency = 1
			btn.Text = "  " .. optStr
			btn.Font = Enum.Font.Gotham
			btn.TextSize = 11
			btn.TextColor3 = KazeUI.CurrentTheme.MutedText
			btn.TextXAlignment = Enum.TextXAlignment.Left
			Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 4)

			local indicator
			if isMulti then
				indicator = Instance.new("Frame", btn)
				indicator.AnchorPoint = Vector2.new(1, 0.5)
				indicator.Position = UDim2.new(1, -8, 0.5, 0)
				indicator.Size = UDim2.fromOffset(8, 8)
				indicator.BackgroundColor3 = KazeUI.CurrentTheme.Background
				Instance.new("UICorner", indicator).CornerRadius = UDim.new(0, 2)
				local indStroke = Instance.new("UIStroke", indicator)
				indStroke.Thickness = 1
				indStroke.Color = KazeUI.CurrentTheme.Border
			end

			local function UpdateVisuals()
				local active = isMulti and selectedMulti[optStr] or (not isMulti and selectedSingle == optStr)
				if active then
					btn.TextColor3 = KazeUI.CurrentTheme.Text
					btn.BackgroundColor3 = KazeUI.CurrentTheme.Background
					if indicator then indicator.BackgroundColor3 = KazeUI.CurrentTheme.Accent end
				else
					btn.TextColor3 = KazeUI.CurrentTheme.MutedText
					btn.BackgroundColor3 = Color3.fromRGB(0,0,0)
					btn.BackgroundTransparency = 1
					if indicator then indicator.BackgroundColor3 = KazeUI.CurrentTheme.Background end
				end
			end
			UpdateVisuals()

			btn.MouseEnter:Connect(function()
				TweenService:Create(btn, TWEEN_FAST, {BackgroundColor3 = KazeUI.CurrentTheme.PanelHover, BackgroundTransparency = 0}):Play()
			end)
			btn.MouseLeave:Connect(function()
				UpdateVisuals()
			end)

			btn.MouseButton1Click:Connect(function()
				if isMulti then
					selectedMulti[optStr] = not selectedMulti[optStr]
					UpdateVisuals()
					UpdateHeader()
					if type(callback) == "function" then
						local r = {}
						for k,v in pairs(selectedMulti) do if v then table.insert(r, k) end end
						task.spawn(callback, r)
					end
				else
					selectedSingle = optStr
					UpdateHeader()
					Close()
					for _, item in ipairs(optionItems) do item.Update() end
					if type(callback) == "function" then task.spawn(callback, selectedSingle) end
				end
			end)

			table.insert(optionItems, { Btn = btn, Text = optStr, Update = UpdateVisuals })
		end
		CalcHeight()
	end
	Refresh(options)

	searchBox:GetPropertyChangedSignal("Text"):Connect(function()
		local q = string.lower(searchBox.Text)
		for _, item in ipairs(optionItems) do
			item.Btn.Visible = (q == "" or string.find(string.lower(item.Text), q, 1, true) ~= nil)
		end
		CalcHeight()
	end)

	clickBtn.MouseButton1Click:Connect(function()
		isOpen = not isOpen
		if isOpen then
			searchWrapper.Visible = true
			listContainer.Visible = true
			CalcHeight()
			TweenService:Create(arrowIcon, TWEEN_FAST, {Rotation = 180}):Play()
			TweenService:Create(boxStroke, TWEEN_FAST, {Color = KazeUI.CurrentTheme.BorderHover}):Play()
		else
			Close()
		end
	end)

	RegisterFlag(flag, function()
		if isMulti then
			local r = {}
			for k,v in pairs(selectedMulti) do if v then table.insert(r, k) end end
			return r
		end
		return selectedSingle
	end, function(v)
		if isMulti then
			selectedMulti = {}
			if type(v) == "table" then for _, val in ipairs(v) do selectedMulti[tostring(val)] = true end end
		else
			selectedSingle = tostring(v)
		end
		UpdateHeader()
		for _, item in ipairs(optionItems) do item.Update() end
	end)

	return {
		Wrapper = wrapper,
		Refresh = function(self, newOpts) Refresh(newOpts) end,
		SetText = function(self, newTitle) titleLabel.Text = newTitle end
	}
end

	valueLabel.Font = Enum.Font.GothamMedium
	valueLabel.TextSize = 11
	valueLabel.TextColor3 = KazeUI.CurrentTheme.MutedText
	valueLabel.TextXAlignment = Enum.TextXAlignment.Right

	local sliderTrack = Instance.new("Frame", card)
	sliderTrack.Size = UDim2.new(1, 0, 0, 4)
	sliderTrack.Position = UDim2.new(0, 0, 0, 22)
	sliderTrack.BackgroundColor3 = KazeUI.CurrentTheme.Background
	Instance.new("UICorner", sliderTrack).CornerRadius = UDim.new(1, 0)

	local sliderFill = Instance.new("Frame", sliderTrack)
	sliderFill.Size = UDim2.fromScale(0, 1)
	sliderFill.BackgroundColor3 = KazeUI.CurrentTheme.Accent
	Instance.new("UICorner", sliderFill).CornerRadius = UDim.new(1, 0)

	local sliderKnob = Instance.new("Frame", sliderTrack)
	sliderKnob.AnchorPoint = Vector2.new(0.5, 0.5)
	sliderKnob.Size = UDim2.fromOffset(10, 10)
	sliderKnob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	Instance.new("UICorner", sliderKnob).CornerRadius = UDim.new(1, 0)

	local knobStroke = Instance.new("UIStroke", sliderKnob)
	knobStroke.Thickness = 1
	knobStroke.Color = KazeUI.CurrentTheme.Border

	local clickDetector = Instance.new("TextButton", sliderTrack)
	clickDetector.Size = UDim2.new(1, 0, 3, 0)
	clickDetector.AnchorPoint = Vector2.new(0, 0.5)
	clickDetector.Position = UDim2.fromScale(0, 0.5)
	clickDetector.BackgroundTransparency = 1
	clickDetector.Text = ""

	local dragging = false
	local function UpdateVal(input)
		local w = sliderTrack.AbsoluteSize.X
		if w <= 0 then return end
		local relX = input.Position.X - sliderTrack.AbsolutePosition.X
		local pct = math.clamp(relX / w, 0, 1)
		local newVal = min + (max - min) * pct
		newVal = math.clamp(math.floor(newVal + 0.5), min, max)
		value = newVal
		valueLabel.Text = tostring(value)

		local displayPct = (max > min) and ((value - min) / (max - min)) or 0
		sliderFill.Size = UDim2.fromScale(displayPct, 1)
		sliderKnob.Position = UDim2.fromScale(displayPct, 0.5)

		if typeof(callback) == "function" then task.spawn(callback, value) end
	end

	local function SetDisplayValue(val)
		value = math.clamp(tonumber(val) or min, min, max)
		valueLabel.Text = tostring(value)
		local displayPct = (max > min) and ((value - min) / (max - min)) or 0
		sliderFill.Size = UDim2.fromScale(displayPct, 1)
		sliderKnob.Position = UDim2.fromScale(displayPct, 0.5)
	end
	SetDisplayValue(value)

	clickDetector.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			ToggleScrollingAncestors(sliderTrack, false)
			LockCamera()
			UpdateVal(input)
			
			local move, stop
			move = UIS.InputChanged:Connect(function(mInput)
				if dragging and (mInput.UserInputType == Enum.UserInputType.MouseMovement or mInput.UserInputType == Enum.UserInputType.Touch) then
					UpdateVal(mInput)
				end
			end)
			stop = UIS.InputEnded:Connect(function(eInput)
				if eInput.UserInputType == Enum.UserInputType.MouseButton1 or eInput.UserInputType == Enum.UserInputType.Touch then
					dragging = false
					ToggleScrollingAncestors(sliderTrack, true)
					UnlockCamera()
					move:Disconnect()
					stop:Disconnect()
				end
			end)
		end
	end)

	RegisterFlag(flag, function() return value end, function(v) SetDisplayValue(v) end)

	return {
		Wrapper = wrapper,
		SetValue = function(self, val) SetDisplayValue(val) end
	}
end

	function Win:CreateConfigManager(tab)
		local sec = tab:SectionUI("Configs")
		local cfgName = "Default"

		local function getConfigs()
			local list = {}
			local folder = KazeUI.AuthorName or "KazeUI"
			if makefolder then pcall(function() makefolder(folder) end) end
			if listfiles then
				local s, files = pcall(function() return listfiles(folder) end)
				if s and type(files) == "table" then
					for _, f in ipairs(files) do
						local fileName = string.match(f, "([^/\\]+)$") or f
						local name = string.match(fileName, "^(.+)%.json$")
						if name then table.insert(list, name) end
					end
				end
			end
			if #list == 0 then table.insert(list, "Default") end
			return list
		end

		sec:TextBox({Title = "Config Name", Default = "Default", Callback = function(v) cfgName = v end})
		
		local dropdown
		sec:Button({Title = "Save Config", Callback = function() 
			KazeUI:SaveConfig(cfgName) 
			if dropdown then dropdown:Refresh(getConfigs()) end
		end})

		dropdown = sec:Dropdown({
			Title = "Select Config",
			Options = getConfigs(),
			Callback = function(v) cfgName = v end
		})

		sec:Button({Title = "Load Config", Callback = function() KazeUI:LoadConfig(cfgName) end})
		sec:Button({Title = "Delete Config", Callback = function()
			if delfile then
				pcall(function() delfile((KazeUI.AuthorName or "KazeUI") .. "/" .. cfgName .. ".json") end)
				if dropdown then dropdown:Refresh(getConfigs()) end
			end
		end})
	end

			txt.TextColor3 = KazeUI.CurrentTheme.Text
			txt.Font = Enum.Font.GothamBold
			if iconLabel then iconLabel.ImageColor3 = KazeUI.CurrentTheme.Accent end
		end

		btn.MouseEnter:Connect(function()
			if not leftIndicator.Visible then
				TweenService:Create(btn, TWEEN_FAST, {BackgroundColor3 = KazeUI.CurrentTheme.PanelHover, BackgroundTransparency = 0}):Play()
				TweenService:Create(txt, TWEEN_FAST, {TextColor3 = KazeUI.CurrentTheme.Text}):Play()
				if iconLabel then TweenService:Create(iconLabel, TWEEN_FAST, {ImageColor3 = KazeUI.CurrentTheme.Text}):Play() end
			end
		end)
		btn.MouseLeave:Connect(function()
			if not leftIndicator.Visible then
				TweenService:Create(btn, TWEEN_FAST, {BackgroundTransparency = 1}):Play()
				TweenService:Create(txt, TWEEN_FAST, {TextColor3 = KazeUI.CurrentTheme.MutedText}):Play()
				if iconLabel then TweenService:Create(iconLabel, TWEEN_FAST, {ImageColor3 = KazeUI.CurrentTheme.MutedText}):Play() end
			end
		end)

		btn.MouseButton1Click:Connect(Activate)
		if #self._Tabs == 1 then Activate() end

		local public = { Button = btn, Page = page }
		AttachElementsToAPI(public, pageContent, 1)
		return public
	end

	table.insert(KazeUI.ActiveWindows, Window)
	return Win
end

return KazeUI
